# 1. 计时器

计时器是异步的，当时机成熟之后才会执行。计时器会返回一个数字，该数字表示计时器的编号。

- **setTimeout** 方法：指定时间到达后运行某个函数
  - **clearTimeout** 方法：清除计时器
- **setInterval** 方法：指定间隔时间到达后运行某个函数
  - **clearInterval** 方法：清除计时器

---

## 1.1 setTimeout - 延时执行

`setTimeout` 在指定毫秒数之后执行一次回调函数。

```html
<button id="btn1">点击开始</button>
<button id="btn2">清除计时器</button>

<script>
    var timer1, timer2, timer3;
    btn1.onclick = function() {
        // 3000毫秒之后，运行指定的函数
        timer1 = setTimeout(function() {
            console.log("计时完成1");
        }, 3000);

        timer2 = setTimeout(function() {
            console.log("计时完成2");
        }, 3000);

        timer3 = setTimeout(function() {
            console.log("计时完成3");
        }, 3000);

        console.log(timer1, timer2, timer3); // 1 2 3 计时器编号
        console.log("点击事件"); // 同步代码先执行
    }

    btn2.onclick = function() {
        clearTimeout(timer1);
        clearTimeout(timer2);
        clearTimeout(timer3);
    }
</script>
```

要点：`setTimeout` 返回一个唯一编号，多个计时器依次编号。计时器是异步的，同步代码（如 `console.log("点击事件")`）会先于计时器回调执行。

---

## 1.2 setInterval - 间隔执行

`setInterval` 每隔指定毫秒数重复执行回调函数。

```html
<h1>0</h1>
<button id="btn1">点击开始</button>
<button id="btn2">清除计时器</button>

<script>
    var timer1, num = 0;
    var h1 = document.querySelector("h1");
    btn1.onclick = function() {
        timer1 = setInterval(function() {
            num++;
            h1.innerHTML = num;
        }, 3000);
    }
    btn2.onclick = function() {
        clearInterval(timer1);
    }
</script>
```

---

## 1.3 用 setTimeout 模拟 setInterval

通过递归调用 `setTimeout` 可以模拟 `setInterval` 的效果，且更灵活（可以在回调中决定是否继续）。

```html
<h1>0</h1>
<button id="btn1">点击开始</button>
<button id="btn2">清除计时器</button>

<script>
    var timer1, num = 0;
    var h1 = document.querySelector("h1");
    btn1.onclick = function() {
        interval(function(){
            num++;
            h1.innerHTML = num;
        }, 300);
    }

    function interval(callback, duration) {
        timer1 = setTimeout(function(){
            callback();
            interval(callback, duration);
        }, duration);
    }

    btn2.onclick = function() {
        clearTimeout(timer1);
    }
</script>
```

## 1.4 文字时钟

使用 `setInterval` 每秒更新时间显示：

```html
<h1></h1>

<script>
    var h1 = document.querySelector("h1");

    function setTime() {
        var date = new Date();
        var hour = date.getHours();
        var minute = date.getMinutes();
        var second = date.getSeconds();
        h1.innerHTML = `${hour.toString().padStart(2,"0")}:${minute.toString().padStart(2,"0")}:${second.toString().padStart(2,"0")}`;
    }

    setTime(); // 初始调用一次，避免空白
    setInterval(setTime, 1000);
</script>
```

要点：使用 `padStart(2, "0")` 补零，确保时间格式为 `HH:MM:SS`。先调用一次 `setTime()` 避免页面初始空白。

## 1.5 自动移动的 div

使用 `setInterval` 实现简单的动画效果，注意防止重复启动计时器：

```html
<style>
    div {
        width: 100px;
        height: 100px;
        background: red;
        position: absolute;
        left: 0;
        top: 0;
    }
</style>

<div></div>
<button id="btn1">开始移动</button>
<button id="btn2">停止移动</button>

<script>
    var div = document.querySelector("div");
    var left = 0;
    var timer;
    btn1.onclick = function() {
        if (timer) {
            return; // 防止重复启动
        }
        timer = setInterval(function() {
            left++;
            div.style.left = left + "px";
        }, 16) // 约60fps
    }

    btn2.onclick = function(){
        clearInterval(timer);
        timer = null; // 重置为null，允许再次启动
    }
</script>
```

要点：16ms 间隔约等于 60fps 的刷新率。通过判断 `timer` 是否存在防止重复启动，清除后将 `timer` 重置为 `null`。

# 2. [作业讲解] 计时器

## 2.1 APM 计算器

APM（Actions Per Minute）计算器：在倒计时内统计点击次数。

```html
<p>倒计时：<span id="spanclock">5</span></p>
<p>点击数：<span id="spannum">0</span></p>
<p>
    <button id="btnadd">点击</button>
    <button id="btnreset">重置</button>
</p>

<script>
    var num = 0,
        clock = 5, // 倒计时秒数
        timer,     // 计时器id
        isOver = false, // 是否结束
        spanNum = document.getElementById("spannum"),
        btnAdd = document.getElementById("btnadd"),
        btnReset = document.getElementById("btnreset"),
        spanClock = document.getElementById("spanclock");

    btnAdd.onclick = function() {
        if (num === 0) {
            setClock(); // 第一次点击时启动倒计时
        }
        if (!isOver) {
            setNum(num + 1);
        }
    }

    btnReset.onclick = function() {
        setNum(0);
        clock = 5;
        spanClock.innerText = clock;
        isOver = false;
    }

    function setNum(n) {
        num = n;
        spanNum.innerText = num;
    }

    // 开启一个倒计时
    function setClock() {
        if (timer) {
            return;
        }
        timer = setInterval(function() {
            clock--;
            spanClock.innerText = clock;
            if (clock === 0) {
                clearInterval(timer);
                timer = null;
                isOver = true; // 结束了
            }
        }, 1000);
    }
</script>
```

要点：第一次点击时才启动倒计时，`isOver` 标志位控制游戏结束后禁止继续点击。

## 2.2 自动移动的 div（带边界反弹）

圆球自动移动，遇到窗口边界自动反弹：

```html
<style>
    .move {
        position: fixed;
        left: 0;
        top: 0;
        width: 100px;
        height: 100px;
        border-radius: 50%;
        background: red;
    }
</style>

<div class="move"></div>

<script>
    var div = document.querySelector(".move"),
        x = 0,       // 初始坐标x
        y = 0,       // 初始坐标y
        xDis = 2,    // 每次移动x的距离
        yDis = 2,    // 每次移动y的距离
        duration = 16, // 移动的间隔时间
        width = 100,
        height = 100;

    setInterval(function() {
        x += xDis;
        y += yDis;

        // 边界判定 - Y轴
        if (y >= document.documentElement.clientHeight - height) {
            y = document.documentElement.clientHeight - height;
            yDis = -yDis; // 反向
        }
        if (y <= 0) {
            y = 0;
            yDis = -yDis;
        }

        // 边界判定 - X轴
        if (x <= 0) {
            x = 0;
            xDis = -xDis;
        }
        if (x >= document.documentElement.clientWidth - width) {
            x = document.documentElement.clientWidth - width;
            xDis = -xDis;
        }

        div.style.left = x + "px";
        div.style.top = y + "px";
    }, duration);
</script>
```

要点：通过取负值（`xDis = -xDis`）实现方向反转，使用 `document.documentElement.clientWidth/clientHeight` 获取视口尺寸进行边界判断。

## 3.3 JavaScript 核心逻辑

无缝轮播图的实现关键在于：在图片列表头部克隆最后一张图片，尾部克隆第一张图片，切换时通过 `marginLeft` 偏移实现动画。

```javascript
// 配置对象
var config = {
    imgWidth: 520,
    dotWidth: 12,
    doms: {
        divBanner: document.querySelector(".banner"),
        divImgs: document.querySelector(".banner .imgs"),
        divDots: document.querySelector(".banner .dots"),
        divArrow: document.querySelector(".banner .arrow")
    },
    currentIndex: 0, // 当前图片索引 0 ~ imgNumber-1
    timer: {
        duration: 16,  // 运动间隔时间(ms)
        total: 300,    // 运动总时间(ms)
        id: null
    },
    autoTimer: null   // 自动轮播计时器
};
config.imgNumber = config.doms.divImgs.children.length;
```

**初始化 - 复制首尾图片实现无缝效果：**

```javascript
function initElements() {
    // 创建小圆点
    for (var i = 0; i < config.imgNumber; i++) {
        var span = document.createElement("span");
        config.doms.divDots.appendChild(span);
    }
    // 复制图片：尾部加第一张，头部加最后一张
    var children = config.doms.divImgs.children;
    var first = children[0], last = children[children.length - 1];
    var newImg = first.cloneNode(true);
    config.doms.divImgs.appendChild(newImg);    // 尾部追加第一张
    newImg = last.cloneNode(true);
    config.doms.divImgs.insertBefore(newImg, first); // 头部插入最后一张
}

function initPosition() {
    var left = (-config.currentIndex - 1) * config.imgWidth;
    config.doms.divImgs.style.marginLeft = left + "px";
}
```

**切换动画核心逻辑：**

```javascript
function switchTo(index, direction) {
    if (index === config.currentIndex) return;
    if (!direction) direction = "left";

    var newLeft = (-index - 1) * config.imgWidth;
    animateSwitch();

    config.currentIndex = index;
    setDotStatus();

    function animateSwitch() {
        stopAnimate();
        var number = Math.ceil(config.timer.total / config.timer.duration);
        var curNumber = 0;
        var distance,
            marginLeft = parseFloat(getComputedStyle(config.doms.divImgs).marginLeft),
            totalWidth = config.imgNumber * config.imgWidth;

        if (direction === "left") {
            if (newLeft < marginLeft) {
                distance = newLeft - marginLeft;
            } else {
                distance = -(totalWidth - Math.abs(newLeft - marginLeft));
            }
        } else {
            if (newLeft > marginLeft) {
                distance = newLeft - marginLeft;
            } else {
                distance = totalWidth - Math.abs(newLeft - marginLeft);
            }
        }

        var everyDistance = distance / number;

        config.timer.id = setInterval(function () {
            marginLeft += everyDistance;
            if (direction === "left" && Math.abs(marginLeft) > totalWidth) {
                marginLeft += totalWidth;
            } else if (direction === "right" && Math.abs(marginLeft) < config.imgWidth) {
                marginLeft -= totalWidth;
            }
            config.doms.divImgs.style.marginLeft = marginLeft + "px";
            curNumber++;
            if (curNumber === number) {
                stopAnimate();
            }
        }, config.timer.duration);
    }

    function stopAnimate() {
        clearInterval(config.timer.id);
        config.timer.id = null;
    }
}
```

**自动轮播与交互：**

```javascript
function toLeft() {
    var index = config.currentIndex - 1;
    if (index < 0) index = config.imgNumber - 1;
    switchTo(index, "right");
}

function toRight() {
    var index = (config.currentIndex + 1) % config.imgNumber;
    switchTo(index, "left");
}

// 自动轮播
config.autoTimer = setInterval(toRight, 2000);

// 鼠标悬停暂停，离开恢复
config.doms.divBanner.onmouseenter = function() {
    clearInterval(config.autoTimer);
    config.autoTimer = null;
}
config.doms.divBanner.onmouseleave = function() {
    if (config.autoTimer) return;
    config.autoTimer = setInterval(toRight, 2000);
}

// 箭头点击
config.doms.divArrow.onclick = function(e) {
    if (e.target.classList.contains("left")) toLeft();
    else if (e.target.classList.contains("right")) toRight();
}

// 小圆点点击
config.doms.divDots.onclick = function(e) {
    if (e.target.tagName === "SPAN") {
        var index = Array.from(this.children).indexOf(e.target);
        switchTo(index, index > config.currentIndex ? "left" : "right");
    }
}
```

# 4. window 对象

## 4.1 自身方法

- **open**：打开一个新窗口

```javascript
window.open("页面路径", "打开目标", "配置")
// 示例
window.open("test2.html", "_self")  // 在当前窗口打开
```

- **alert**：弹出警告框
- **confirm**：弹出确认框
- **prompt**：弹出输入框

## 4.2 对象属性

### document

```javascript
document.write("内容")
```

`document.write` 在当前文档流中写入内容，如果当前文档流不存在，则新开一个文档流。

### location - 地址栏对象

- **href** 属性：获取或设置当前地址

```javascript
console.log(location.href);  // 获取当前地址
location.href = "http://example.com";  // 跳转
```

- **reload** 方法：刷新当前页面

```javascript
location.reload();
```

### navigator - 浏览器信息对象

用于获取浏览器相关信息（如 userAgent）。

### history - 历史记录

```javascript
history.go(-1);      // 后退一步
history.back();      // 后退
history.forward();   // 前进
```

### console

```javascript
console.log();       // 打印对象的valueOf的返回值
console.dir();       // 打印对象结构
console.time("标签");   // 开始计时
console.timeEnd("标签"); // 结束计时
```

**console.time / timeEnd 计时示例：**

```javascript
console.time("全局");
for (var i = 0; i < 100000; i++) {
}

console.time("测试");
var sum = 0;
for (var i = 0; i < 10000; i++) {
    sum += i;
}
console.timeEnd("测试");
console.timeEnd("全局");
```

# 5. [作业讲解] 提示插件

## 5.1 插件的特点

1. **通用性** - 可以在不同项目中复用
2. **易用性** - 接口简单，方便调用
3. **尽量不要与其他功能冲突** - 避免全局污染、命名冲突

## 5.2 作业要求

实现点击弹出提示（confirm）的功能，封装为插件形式：

```html
<button>打开一个确认对话框</button>

<script src="../../plugin/confirm.js"></script>
<script>
    document.querySelector("button").onclick = function() {
        // 打开一个确认对话框
        myPlugin.openConfirm("确认提示内容");
    }
</script>
```

插件将确认对话框封装为独立的 JS 文件，通过 `myPlugin.openConfirm()` 调用，体现了插件的通用性和易用性。

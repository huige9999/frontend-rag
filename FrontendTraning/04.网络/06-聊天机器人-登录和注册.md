## 1. 表单验证器设计模式

### 1.1 验证器基础类

在用户登录和注册功能中，我们需要对表单进行验证。为了代码复用，我们设计了一个通用的表单验证器类。

```javascript
/**
 * 对某一个表单项进行验证的构造函数
 */
class FieldValidator {
  /**
   * 构造器
   * @param {String} txtId 文本框的Id
   * @param {Function} validatorFunc 验证规则函数，当需要对该文本框进行验证时，会调用该函数，函数的参数为当前文本框的值，函数的返回值为验证的错误消息，若没有返回，则表示无错误
   */
  constructor(txtId, validatorFunc) {
    this.input = $('#' + txtId);          // 获取文本框DOM元素
    this.p = this.input.nextElementSibling; // 获取错误提示元素
    this.validatorFunc = validatorFunc;    // 保存验证函数
    
    // 当文本框失去焦点时，自动进行验证
    this.input.onblur = () => {
      this.validate();
    };
  }

  /**
   * 验证，成功返回true，失败返回false
   */
  async validate() {
    const err = await this.validatorFunc(this.input.value);
    if (err) {
      // 有错误
      this.p.innerText = err;  // 显示错误信息
      return false;
    } else {
      this.p.innerText = '';   // 清空错误信息
      return true;
    }
  }

  /**
   * 对传入的所有验证器进行统一的验证，如果所有的验证均通过，则返回true，否则返回false
   * @param {FieldValidator[]} validators
   */
  static async validate(...validators) {
    const proms = validators.map((v) => v.validate()); // 并行验证所有字段
    const results = await Promise.all(proms);         // 等待所有验证完成
    return results.every((r) => r);                    // 所有验证都通过才返回true
  }
}
```

**设计要点：**

1. **单一职责**：每个验证器只负责一个表单项的验证
2. **开闭原则**：通过传入不同的验证函数来扩展验证规则
3. **异步支持**：支持异步验证（如检查用户名是否已存在）
4. **自动触发**：失焦时自动验证，提升用户体验

### 2.2 注册验证逻辑

```javascript
// 账号验证器 - 包含异步验证
const loginIdValidator = new FieldValidator('txtLoginId', async function (val) {
  if (!val) {
    return '请填写账号';
  }
  // 异步检查账号是否已存在
  const resp = await API.exists(val);
  if (resp.data) {
    return '该账号已被占用，请重新选择一个账号名';
  }
});

// 昵称验证器
const nicknameValidator = new FieldValidator('txtNickname', function (val) {
  if (!val) {
    return '请填写昵称';
  }
});

// 密码验证器
const loginPwdValidator = new FieldValidator('txtLoginPwd', function (val) {
  if (!val) {
    return '请填写密码';
  }
});

// 确认密码验证器
const loginPwdConfirmValidator = new FieldValidator(
  'txtLoginPwdConfirm',
  function (val) {
    if (!val) {
      return '请填写确认密码';
    }
    // 验证两次密码是否一致
    if (val !== loginPwdValidator.input.value) {
      return '两次密码不一致';
    }
  }
);

const form = $('.user-form');

// 表单提交处理
form.onsubmit = async function (e) {
  e.preventDefault(); // 阻止表单默认提交行为
  
  // 验证所有字段
  const result = await FieldValidator.validate(
    loginIdValidator,
    nicknameValidator,
    loginPwdValidator,
    loginPwdConfirmValidator
  );
  
  if (!result) {
    return; // 验证未通过，不提交
  }
  
  // 获取表单数据
  const formData = new FormData(form);
  const data = Object.fromEntries(formData.entries());

  // 调用注册API
  const resp = await API.reg(data);
  if (resp.code === 0) {
    alert('注册成功，点击确定，跳转到登录页');
    location.href = './login.html';
  }
};
```

**关键点说明：**

1. **异步验证**：账号验证使用异步函数检查用户名是否已存在
2. **密码确认**：需要访问另一个验证器的input值来比较密码
3. **FormData API**：使用FormData和Object.fromEntries便捷地获取表单数据
4. **阻止默认行为**：使用preventDefault阻止表单默认提交

## 4. API封装与Token认证

### 4.1 API模块设计

```javascript
var API = (function () {
  const BASE_URL = 'http://localhost:7001';  // API基础地址
  const TOKEN_KEY = 'token';                 // localStorage中token的键名

  /**
   * GET请求封装
   * @param {String} path 请求路径
   */
  function get(path) {
    const headers = {};
    // 从localStorage获取token
    const token = localStorage.getItem(TOKEN_KEY);
    if (token) {
      // 将token添加到请求头
      headers.authorization = `Bearer ${token}`;
    }
    return fetch(BASE_URL + path, { headers });
  }

  /**
   * POST请求封装
   * @param {String} path 请求路径
   * @param {Object} bodyObj 请求体对象
   */
  function post(path, bodyObj) {
    const headers = {
      'Content-Type': 'application/json',
    };
    // 从localStorage获取token
    const token = localStorage.getItem(TOKEN_KEY);
    if (token) {
      // 将token添加到请求头
      headers.authorization = `Bearer ${token}`;
    }
    return fetch(BASE_URL + path, {
      headers,
      method: 'POST',
      body: JSON.stringify(bodyObj), // 将对象转换为JSON字符串
    });
  }

  /**
   * 用户注册
   * @param {Object} userInfo 用户信息 {loginId, nickname, loginPwd}
   */
  async function reg(userInfo) {
    const resp = await post('/api/user/reg', userInfo);
    return await resp.json();
  }

  /**
   * 用户登录
   * @param {Object} loginInfo 登录信息 {loginId, loginPwd}
   */
  async function login(loginInfo) {
    const resp = await post('/api/user/login', loginInfo);
    const result = await resp.json();
    
    if (result.code === 0) {
      // 登录成功，将响应头中的token保存到localStorage
      const token = resp.headers.get('authorization');
      localStorage.setItem(TOKEN_KEY, token);
    }
    return result;
  }

  /**
   * 检查账号是否存在
   * @param {String} loginId 账号
   */
  async function exists(loginId) {
    const resp = await get('/api/user/exists?loginId=' + loginId);
    return await resp.json();
  }

  /**
   * 获取用户信息
   */
  async function profile() {
    const resp = await get('/api/user/profile');
    return await resp.json();
  }

  /**
   * 发送聊天消息
   * @param {String} content 消息内容
   */
  async function sendChat(content) {
    const resp = await post('/api/chat', {
      content,
    });
    return await resp.json();
  }

  /**
   * 获取聊天历史
   */
  async function getHistory() {
    const resp = await get('/api/chat/history');
    return await resp.json();
  }

  /**
   * 登出（清除token）
   */
  function loginOut() {
    localStorage.removeItem(TOKEN_KEY);
  }

  // 暴露公共接口
  return {
    reg,
    login,
    exists,
    profile,
    sendChat,
    getHistory,
    loginOut,
  };
})();
```

### 4.2 Token认证机制说明

**什么是Token？**

Token（令牌）是用户身份验证的凭证，类似于一张"通行证"。

**Token认证流程：**

1. **登录获取Token**
   - 用户提交账号密码
   - 服务器验证成功后返回Token
   - 客户端将Token保存到localStorage

2. **携带Token请求**
   - 后续请求从localStorage读取Token
   - 将Token添加到请求头的`authorization`字段
   - 格式：`Bearer ${token}`

3. **服务器验证**
   - 服务器从请求头获取Token
   - 验证Token有效性
   - 根据验证结果返回相应数据

**Token存储位置：**

```javascript
// 保存Token
localStorage.setItem('token', tokenValue);

// 读取Token
const token = localStorage.getItem('token');

// 删除Token（登出）
localStorage.removeItem('token');
```

**优点：**

- 无状态：服务器不需要存储会话信息
- 可扩展：适合分布式系统
- 跨域：支持跨域请求

## 10. 常见问题

### Q1: 为什么要使用Token而不是Session？

**A:** Token具有以下优势：
- **无状态**：服务器不需要存储会话信息
- **可扩展**：适合分布式系统和微服务架构
- **跨域支持**：更适合前后端分离的项目
- **移动端友好**：移动应用更方便使用Token

### Q2: 如何保证Token安全？

**A:** 安全措施包括：
1. 使用HTTPS传输
2. Token设置有效期
3. 使用加密算法生成Token
4. 敏感操作需要二次验证

### Q3: 为什么需要异步验证？

**A:** 异步验证用于：
1. 检查用户名是否已存在（需要查询数据库）
2. 验证码验证（需要调用验证服务）
3. 密码强度检查（可能需要调用API）
4. 实时搜索建议（需要查询后端）

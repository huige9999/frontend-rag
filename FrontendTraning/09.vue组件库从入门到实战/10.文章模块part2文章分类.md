```vue
  getBlogType,
  addBlogType,
  delBlogType,

  updateOneBlogType,

      select: "1",

      data: [],
      dialogFormVisible: false,

        order: ''

        this.dialogFormVisible = true;

          confirmButtonText: "确定",
          cancelButtonText: "取消",
          type: "warning",
        }
      )
        .then(() => {
          delBlogType(id).then(() => {
            this.fetchData();
            this.$message.success("删除成功");
          });
        })
        .catch(() => {
          this.$message({
            type: "info",
            message: "已取消删除",
          });
        });
    },
  },
};
</script>
```

### 2. 封装文章分类 API

创建 `src/api/blogType.js`：

```javascript
import request from '@/utils/request'

// 查看文章分类接口
export function getBlogType() {
  return request.get('/api/blogtype');
}

// 添加文章分类接口
export function addBlogType(data) {
  return request({
    url: '/api/blogtype',
    method: 'post',
    data
  })
}

// 删除文章分类接口
export function delBlogType(id) {
  return request.delete(`/api/blogtype/${id}`);
}

// 查找某一个文章分类
export function findOneBlogType(id) {
  return request({
    url: `/api/blogtype/${id}`,
    method: 'get',
  })
}

// 更新某一个文章分类
export function updateOneBlogType(editInfo) {
  return request({
    url: `/api/blogtype/${editInfo.id}`,
    method: 'put',
    data: editInfo.data
  })
}
```

## 技术要点解析

### 1. Element UI 选择器组件

**el-select** 和 **el-option** 的使用：

```vue
<el-select v-model="select" placeholder="请选择">
  <el-option label="1" value="1"></el-option>
  <el-option label="2" value="2"></el-option>
</el-select>
```

- **v-model**：绑定的值
- **placeholder**：占位符
- **label**：显示的文本
- **value**：实际的值

### 2. 带插槽的输入框

**el-input** 的插槽使用：

```vue
<el-input v-model="input" class="input-with-select">
  <el-select v-model="select" slot="prepend" placeholder="请选择">
    <!-- 选项 -->
  </el-select>
</el-input>
```

- **slot="prepend"**：前置插槽
- **slot="append"**：后置插槽

### 3. CRUD 操作流程

#### 创建 (Create)

```javascript
// 1. 收集表单数据
const formData = { name: this.input, order: this.select };

// 2. 调用接口
addBlogType(formData).then(() => {
  // 3. 刷新列表
  this.fetchData();
  // 4. 提示成功
  this.$message.success("添加分类成功");
});
```

#### 读取 (Read)

```javascript
getBlogType().then((res) => {
  this.data = res.data;
});
```

#### 更新 (Update)

```javascript
// 1. 获取单条数据回填
findOneBlogType(id).then(res => {
  this.form = res.data;
  this.dialogFormVisible = true;
});

// 2. 提交更新
updateOneBlogType({
  id: this.form.id,
  data: this.form
}).then(() => {
  this.fetchData();
});
```

#### 删除 (Delete)

```javascript
// 1. 确认删除
this.$confirm("确认删除吗？", "提示", {
  type: "warning"
})
.then(() => {
  // 2. 调用删除接口
  delBlogType(id).then(() => {
    this.fetchData();
  });
})
```

### 4. 表单数据回填


### 5. 对象解构赋值

```javascript
// 从数组对象中解构出 id
editBlogTypeHandle({id}) {
  // 直接使用 id
  findOneBlogType(id);
}

// 等同于
editBlogTypeHandle(blogType) {
  const id = blogType.id;
  findOneBlogType(id);
}
```

### 6. 条件判断

## 表单验证优化

### 1. 基础验证

```javascript
addBlogTypeHandle() {
  if (!this.input) {
    this.$message.error("分类名称不能为空");
    return;
  }
  
  if (!this.select) {
```

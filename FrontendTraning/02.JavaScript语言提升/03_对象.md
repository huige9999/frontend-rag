# 03. 对象

## 1. 对象基础

### 1.1 对象字面量

对象是JavaScript中最重要的数据结构之一，用于存储键值对数据。

```javascript
// 创建对象的基本方式
const person = {
  name: '张三',
  age: 25,
  gender: '男'
};

console.log(person.name);    // 访问属性：点语法
console.log(person['age']);  // 访问属性：方括号语法
```

### 1.2 对象属性

- **属性名**：可以是字符串或Symbol
- **属性值**：可以是任意类型的值
- **访问方式**：点语法（`.`）和方括号语法（`[]`）

```javascript
const obj = {
  firstName: '张',
  'last name': '三',  // 属性名包含空格，必须用引号
  123: '数字键'       // 数字键会被转换为字符串
};

// 点语法：属性名必须是合法的标识符
console.log(obj.firstName);

// 方括号语法：支持任意字符串和表达式
console.log(obj['last name']);
console.log(obj[123]);
const key = 'firstName';
console.log(obj[key]);  // 通过变量访问
```

## 2. 对象方法

### 2.1 方法定义

对象中的函数称为方法。

```javascript
const user = {
  name: 'monica',
  age: 17,
  
  // 方法简写（ES6）
  sayHello() {
    console.log(`你好，我是${this.name}`);
  },
  
  // 完整写法
  introduce: function() {
    console.log(`我今年${this.age}岁`);
  }
};

user.sayHello();     // 你好，我是monica
user.introduce();    // 我今年17岁
```

### 2.2 this关键字

- `this`指向方法的调用者
- 箭头函数没有自己的`this`，会继承外层作用域的`this`

```javascript
const person = {
  name: 'Alice',
  friends: ['Bob', 'Charlie'],
  
  printFriends() {
    this.friends.forEach(friend => {
      console.log(`${this.name}的朋友：${friend}`);
    });
  }
};

person.printFriends();
// Alice的朋友：Bob
// Alice的朋友：Charlie
```

## 3. 对象操作

### 3.1 属性的增删改查

```javascript
const obj = { a: 1 };

// 添加属性
obj.b = 2;
obj['c'] = 3;

// 修改属性
obj.a = 10;

// 删除属性
delete obj.b;

// 检查属性是否存在
console.log('a' in obj);      // true
console.log(obj.hasOwnProperty('a')); // true
```

### 3.2 对象遍历

```javascript
const user = {
  name: '张三',
  age: 25,
  city: '北京'
};

// 1. for...in 遍历属性名
for (let key in user) {
  console.log(key);  // name, age, city
}

// 2. Object.keys() 获取所有属性名
const keys = Object.keys(user);  // ['name', 'age', 'city']
keys.forEach(key => console.log(key));

// 3. Object.values() 获取所有属性值
const values = Object.values(user);  // ['张三', 25, '北京']

// 4. Object.entries() 获取所有键值对
const entries = Object.entries(user);  // [['name', '张三'], ['age', 25], ['city', '北京']]
for (let [key, value] of entries) {
  console.log(`${key}: ${value}`);
}
```

## 4. 对象的高级操作

### 4.1 对象解构

```javascript
// 基本解构
const person = {
  firstName: '张',
  lastName: '三',
  age: 25
};

const { firstName, lastName } = person;
console.log(firstName, lastName);  // 张 三

// 解构重命名
const { firstName: first, lastName: last } = person;
console.log(first, last);  // 张 三

// 嵌套解构
const user = {
  name: 'Alice',
  address: {
    city: '北京',
    street: '朝阳路'
  }
};
const { address: { city, street } } = user;
console.log(city, street);  // 北京 朝阳路

// 默认值
const { name, age = 18 } = person;
console.log(age);  // 18
```

### 4.2 对象展开

```javascript
const obj = { a: 1, b: 2, c: 3 };

// 复制对象
const obj1 = { ...obj };
console.log(obj1);  // { a: 1, b: 2, c: 3 }

// 合并对象
const obj2 = { ...obj, d: 4 };
console.log(obj2);  // { a: 1, b: 2, c: 3, d: 4 }

// 覆盖属性
const obj3 = { ...obj, a: 10 };
console.log(obj3);  // { a: 10, b: 2, c: 3 }

// 排除属性（使用解构）
const { a, ...rest } = obj;
console.log(rest);  // { b: 2, c: 3 }
```

### 4.3 对象合并与默认值

```javascript
// 创建默认选项对象
function createOptions(options = {}) {
  const defaults = {
    time: 1000,
    speed: 50,
    text: ''
  };
  
  return { ...defaults, ...options };
}

console.log(createOptions());  // { time: 1000, speed: 50, text: '' }
console.log(createOptions({ time: 500 }));  // { time: 500, speed: 50, text: '' }
console.log(createOptions({ 
  time: 500, 
  text: 'hello world' 
}));  // { time: 500, speed: 50, text: 'hello world' }
```

### 4.4 属性描述符

```javascript
const obj = { a: 1 };

// 获取属性描述符
const descriptor = Object.getOwnPropertyDescriptor(obj, 'a');
console.log(descriptor);
// { value: 1, writable: true, enumerable: true, configurable: true }

// 定义属性
Object.defineProperty(obj, 'b', {
  value: 2,
  writable: false,    // 不可写
  enumerable: true,   // 可枚举
  configurable: false // 不可配置
});

// 批量定义属性
Object.defineProperties(obj, {
  c: { value: 3, writable: true },
  d: { value: 4, writable: false }
});
```

### 4.5 Getter和Setter

```javascript
const user = {
  _name: 'monica',
  _age: 17,
  
  // Getter：读取属性时调用
  get name() {
    console.log(`正在读取name属性，值为${this._name}`);
    return this._name;
  },
  
  // Setter：设置属性时调用
  set name(newValue) {
    console.log(`正在设置name属性，新的值为${newValue}`);
    this._name = newValue;
  },
  
  get age() {
    console.log(`正在读取age属性，值为${this._age}`);
    return this._age;
  },
  
  set age(newValue) {
    console.log(`正在设置age属性，新的值为${newValue}`);
    this._age = newValue;
  }
};

user.name;           // 正在读取name属性，值为monica
user.name = 'Alice'; // 正在设置name属性，新的值为Alice
user.age = 18;       // 正在设置age属性，新的值为18
```

## 5. 对象的常用方法

### 5.1 Object静态方法

```javascript
const obj = { a: 1, b: 2, c: 3 };

// Object.keys()：获取所有属性名
console.log(Object.keys(obj));  // ['a', 'b', 'c']

// Object.values()：获取所有属性值
console.log(Object.values(obj));  // [1, 2, 3]

// Object.entries()：获取所有键值对
console.log(Object.entries(obj));  // [['a', 1], ['b', 2], ['c', 3]]

// Object.assign()：合并对象
const target = { a: 1 };
const source = { b: 2, c: 3 };
Object.assign(target, source);
console.log(target);  // { a: 1, b: 2, c: 3 }

// Object.fromEntries()：从键值对数组创建对象
const entries = [['name', 'Alice'], ['age', 25]];
const person = Object.fromEntries(entries);
console.log(person);  // { name: 'Alice', age: 25 }

// Object.freeze()：冻结对象（不可修改）
const frozen = Object.freeze({ a: 1 });
frozen.a = 2;  // 严格模式下会报错
console.log(frozen.a);  // 1

// Object.seal()：密封对象（不可添加删除属性，但可修改）
const sealed = Object.seal({ a: 1 });
sealed.a = 2;  // 可以修改
sealed.b = 3;  // 严格模式下会报错
console.log(sealed);  // { a: 2 }
```

### 5.2 对象原型方法

```javascript
const obj = { a: 1, b: 2 };

// hasOwnProperty()：检查自身属性
console.log(obj.hasOwnProperty('a'));  // true
console.log(obj.hasOwnProperty('toString'));  // false

// propertyIsEnumerable()：检查属性是否可枚举
console.log(obj.propertyIsEnumerable('a'));  // true

// toString()：转换为字符串
console.log(obj.toString());  // [object Object]

// valueOf()：返回原始值
const num = new Number(123);
console.log(num.valueOf());  // 123
```

## 6. 实用技巧

### 6.1 对象去重

```javascript
// 利用对象属性唯一性实现数组去重
const arr = [1, 4, 7, 1, 2, 3, 3, 6, 8];

function uniqueArray(arr) {
  const result = [];
  const seen = {};
  
  for (let item of arr) {
    if (!seen[item]) {
      seen[item] = true;
      result.push(item);
    }
  }
  
  return result;
}

console.log(uniqueArray(arr));  // [1, 4, 7, 2, 3, 6, 8]
```

### 6.2 动态属性名

```javascript
const key = 'name';
const value = 'Alice';

// 计算属性名
const obj = {
  [key]: value,
  ['user_' + key]: value
};

console.log(obj);  // { name: 'Alice', user_name: 'Alice' }
```

### 6.3 条件属性

```javascript
const isVIP = true;
const age = 25;

const user = {
  name: 'Alice',
  ...(isVIP && { level: 'VIP' }),  // 条件为true时添加
  ...(age >= 18 && { isAdult: true })
};

console.log(user);  // { name: 'Alice', level: 'VIP', isAdult: true }
```

### 练习4：实现Getter和Setter

为对象的所有属性添加getter和setter：

```javascript
function createObservableObject(obj) {
  const result = {};
  
  for (let key in obj) {
    const value = obj[key];
    const privateKey = '_' + key;
    result[privateKey] = value;
    
    Object.defineProperty(result, key, {
      get() {
        console.log(`正在读取${key}属性，值为${this[privateKey]}`);
        return this[privateKey];
      },
      set(newValue) {
        console.log(`正在设置${key}属性，新的值为${newValue}`);
        this[privateKey] = newValue;
      }
    });
  }
  
  return result;
}

const user = createObservableObject({
  name: 'monica',
  age: 17,
  sex: 'female'
});

user.name;           // 正在读取name属性，值为monica
user.age = 18;       // 正在设置age属性，新的值为18
console.log(user.sex); // 正在读取sex属性，值为female
```

### 练习5：数组去重

使用对象实现数组去重：

```javascript
const arr = [1, 4, 7, 1, 2, 3, 3, 6, 8];

// 方法1：使用对象记录
function uniqueArray1(arr) {
  const seen = {};
  const result = [];
  
  for (let item of arr) {
    if (!seen[item]) {
      seen[item] = true;
      result.push(item);
    }
  }
  
  return result;
}

// 方法2：使用Set
function uniqueArray2(arr) {
  return [...new Set(arr)];
}

// 方法3：使用filter
function uniqueArray3(arr) {
  return arr.filter((item, index) => arr.indexOf(item) === index);
}

console.log(uniqueArray1(arr));  // [1, 4, 7, 2, 3, 6, 8]
console.log(uniqueArray2(arr));  // [1, 4, 7, 2, 3, 6, 8]
console.log(uniqueArray3(arr));  // [1, 4, 7, 2, 3, 6, 8]
```

## 8. 总结

### 8.1 核心概念

- **对象**：键值对集合，JavaScript最核心的数据结构
- **属性**：对象的特征，包含属性名和属性值
- **方法**：对象的函数，作为属性存储
- **this**：指向方法的调用者

### 8.2 常用操作

| 操作 | 方法 |
|------|------|
| 创建对象 | `{}`, `new Object()` |
| 访问属性 | `obj.key`, `obj[key]` |
| 添加属性 | `obj.newKey = value` |
| 删除属性 | `delete obj.key` |
| 遍历对象 | `for...in`, `Object.keys()`, `Object.entries()` |
| 复制对象 | `{...obj}`, `Object.assign()` |
| 合并对象 | `{...obj1, ...obj2}` |

### 8.3 最佳实践

1. 优先使用对象字面量创建对象
2. 使用解构提取需要的属性
3. 使用展开运算符合并和复制对象
4. 合理使用getter/setter控制属性访问
5. 避免直接修改传入的对象参数
6. 使用`Object.freeze()`保护不可变对象

### 8.4 注意事项

- 对象是引用类型，赋值时传递引用
- `for...in`会遍历原型链上的可枚举属性
- 使用`hasOwnProperty`检查自身属性
- 箭头函数没有自己的`this`
- 注意属性名的计算和动态生成

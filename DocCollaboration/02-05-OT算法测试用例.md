# OT算法测试用例

> 本节课主要通过完整的测试用例来验证OT算法的正确性，包括操作应用、撤销操作以及协同编辑的模拟。

## 关键代码

### index.js

入口文件，演示了操作的创建与应用、撤销操作，以及两个客户端之间协同编辑的完整流程。

```js
const { sampleText } = require("./data/sampleText");
const { Client } = require("./Client");
const { TextOperation } = require("./ot/TextOperation");

// 先来模拟撤销操作
const base = sampleText; // 基础文本
const op = new TextOperation(); // 创建一个操作对象
op.retain(5).delete(5).insert("Universe"); // 执行一系列的操作
const newText = op.apply(base);
console.log("更新后的文本内容：", newText); // 输出操作后的文本

// 接下来进行撤销的测试
const inversedOp = op.invert(base); // 反转操作对象
const restoredText = inversedOp.apply(newText); // 应用反转操作对象
console.log("撤销后的文本内容：", restoredText); // 输出撤销后的文本

console.log("===============================");

// 下面测试协同编辑的操作
const A = new Client("A"); // 创建客户端 A
const B = new Client("B"); // 创建客户端 B

A.doc = B.doc = sampleText; // 初始化客户端的文档内容

// 首先是客户端 A 进行本地操作
const opA = new TextOperation().retain(10).insert("X"); // 在原本的文本后面插入一个 X
A.applyLocal(opA); // 客户端A执行本地操作
B.receiveRemote({ version: 0, op: opA }); // 客户端B接收远程操作

console.log("===============================");
// 接下来是客户端 B 进行本地操作
const opB = new TextOperation().retain(10).insert("Y"); // 在原本的文本后面插入一个 Y
B.applyLocal(opB); // 客户端B执行本地操作
A.receiveRemote({ version: 1, op: opB }); // 客户端A接收远程操作
```

### Client/index.js

Client 类模拟了协同编辑中客户端的行为，包含本地操作应用和远程操作接收两个核心方法。当本地有未提交的操作时，远程操作到达后需要通过 Transform 进行转换再应用。

```js
const { transform } = require("../ot/Transform"); // 引入 transform 函数

class Client {
  constructor(name) {
    this.name = name; // 客户端的名称
    this.version = 0; // 当前客户端的版本号
    this.buffer = []; // 用于存储本地的操作
    this.doc = ""; // 用于存储当前客户端的文档内容
  }

  /**
   * 本地执行 op 里面的 ops 操作对象列表的所有操作
   * @param {*} op 接收一个 TextOperation 实例对象
   */
  applyLocal(op) {
    this.doc = op.apply(this.doc); // 执行操作，将操作后的结果更新到 doc 中
    this.buffer.push({ version: this.version, op }); // 将当前的操作对象推入到 buffer 中
    console.log(
      `客户端 ${this.name} 执行了本地操作，当前的文档内容为：${this.doc}`
    );
  }

  /**
   * 接收远程服务器传递过来的操作对象
   * @param {*} remoteOp 包含操作对象以及版本号的对象
   */
  receiveRemote(remoteOp) {
    if (this.buffer.length === 0) {
      // 本地没有未提交的更新，直接应用远程操作
      this.doc = remoteOp.op.apply(this.doc);
      console.log(
        `客户端 ${this.name} 执行了远程操作，合并之后当前的文档内容为：${this.doc}`
      );
    } else {
      // 本地和远程都有更新，需要进行转换
      const [remotePrime, localPrime] = transform(
        remoteOp.op, // 别人（服务器）发过来的操作对象
        this.buffer[0].op // 本地还没有提交的操作对象
      );

      // remotePrime：转换后的远程操作对象，可以安全的应用到当前的文档内容上
      // localPrime：转换后的本地操作对象，已经调整好了位置
      this.doc = remotePrime.apply(this.doc); // 执行转换后的远程操作对象
      this.buffer[0].op = localPrime; // 更新本地缓存的操作对象

      console.log(
        `客户端 ${this.name} 执行了远程操作，合并之后当前的文档内容为：${this.doc}`
      );
    }

    this.version++; // 更新版本号
  }
}

module.exports = {
  Client,
};
```

### ot/TextOperation.js

TextOperation 操作类，提供 retain、insert、delete、apply、invert、toJSON 等方法，与前面课程实现一致。

```js
class TextOperation {
  constructor() {
    this.ops = [];
  }

  retain(n) {
    if (n > 0) {
      this.ops.push({ type: "retain", count: n });
    }
    return this;
  }

  insert(str) {
    if (str && str.length > 0) {
      this.ops.push({ type: "insert", value: str });
    }
    return this;
  }

  delete(n) {
    if (n > 0) {
      this.ops.push({ type: "delete", count: n });
    }
    return this;
  }

  apply(text) {
    let result = "";
    let index = 0;
    for (const op of this.ops) {
      if (op.type === "retain") {
        result += text.slice(index, index + op.count);
        index += op.count;
      } else if (op.type === "insert") {
        result += op.value;
      } else if (op.type === "delete") {
        index += op.count;
      }
    }
    result += text.slice(index);
    return result;
  }

  invert(baseText) {
    const inverse = new TextOperation();
    let index = 0;
    for (const op of this.ops) {
      if (op.type === "retain") {
        inverse.retain(op.count);
        index += op.count;
      } else if (op.type === "insert") {
        inverse.delete(op.value.length);
      } else if (op.type === "delete") {
        const deletedText = baseText.slice(index, index + op.count);
        inverse.insert(deletedText);
        index += op.count;
      }
    }
    return inverse;
  }

  toJSON() {
    return this.ops;
  }
}

module.exports = {
  TextOperation,
};
```

### ot/Transform.js

Transform 转换函数，处理所有操作组合的 case-by-case 转换规则。

```js
const { TextOperation } = require("./TextOperation");

function transform(opA, opB) {
  const a = opA.toJSON();
  const b = opB.toJSON();

  const aPrime = new TextOperation();
  const bPrime = new TextOperation();

  let i = 0,
    j = 0;

  while (i < a.length || j < b.length) {
    const op1 = a[i] || null;
    const op2 = b[j] || null;

    if (op1 && op1.type === "insert") {
      aPrime.insert(op1.value);
      bPrime.retain(op1.value.length);
      i++;
      continue;
    }

    if (op2 && op2.type === "insert") {
      bPrime.insert(op2.value);
      aPrime.retain(op2.value.length);
      j++;
      continue;
    }

    if (op1 && op1.type === "retain" && op2 && op2.type === "retain") {
      const minCount = Math.min(op1.count, op2.count);
      aPrime.retain(minCount);
      bPrime.retain(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    if (op1 && op1.type === "delete" && op2 && op2.type === "retain") {
      const minCount = Math.min(op1.count, op2.count);
      aPrime.delete(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    if (op1 && op1.type === "retain" && op2 && op2.type === "delete") {
      const minCount = Math.min(op1.count, op2.count);
      bPrime.delete(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    if (op1 && op1.type === "delete" && op2 && op2.type === "delete") {
      const minCount = Math.min(op1.count, op2.count);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    if (op1 && op1.type === "insert" && op2 && op2.type === "delete") {
      aPrime.insert(op1.value);
      bPrime.retain(op1.value.length);
      i++;
      continue;
    }

    if (op1 && op1.type === "delete" && op2 && op2.type === "insert") {
      bPrime.insert(op2.value);
      aPrime.retain(op2.value.length);
      j++;
      continue;
    }

    throw new Error("未知的操作组合");
  }

  return [aPrime, bPrime];
}

module.exports = {
  transform,
};
```

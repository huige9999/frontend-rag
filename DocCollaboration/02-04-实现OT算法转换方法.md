# 实现OT算法转换方法

> 本节课主要通过代码实现来学习OT算法中转换（Transform）方法的设计与实现。

## 关键代码

### Transform.js

Transform 函数是 OT 算法的核心，负责将两个并发的操作进行转换，确保两个操作在对方的文档上应用后最终结果一致。该函数通过 case-by-case 的规则匹配，处理所有可能的操作组合（insert/retain/delete）。

```js
const { TextOperation } = require("./TextOperation");
/**
 *
 * @param {*} opA 用户A的 TextOperation 对象
 * @param {*} opB 用户B的 TextOperation 对象
 */
function transform(opA, opB) {
  // 首先获取到用户 A 和用户 B 的操作描述对象列表
  const a = opA.toJSON(); // 用户 A 的操作描述对象列表
  const b = opB.toJSON(); // 用户 B 的操作描述对象列表

  // 之所以要实例化新的 TextOperation 对象，是因为无论是用户 A，还是用户 B，
  // 需要对它们的操作描述对象进行一个转换
  const aPrime = new TextOperation(); // 存储用户 A 转换后的操作描述对象
  const bPrime = new TextOperation(); // 存储用户 B 转换后的操作描述对象

  let i = 0,
    j = 0; // 这是两个指针，或者理解为下标值，用于遍历操作描述对象列表

  // 遍历两个操作列表，逐步处理所有的操作
  // 这个循环实际上就是一个 case-by-case 的规则匹配过程
  while (i < a.length || j < b.length) {
    const op1 = a[i] || null; // 取出当前用户 A 操作列表中的一个操作描述对象
    const op2 = b[j] || null; // 取出当前用户 B 操作列表中的一个操作描述对象

    // 情况1：A 要插入，B 需要保留 A 插入的长度来绕开
    if (op1 && op1.type === "insert") {
      aPrime.insert(op1.value);
      bPrime.retain(op1.value.length);
      i++;
      continue;
    }

    // 情况2：B 要插入，A 需要保留 B 插入的长度来绕开
    if (op2 && op2.type === "insert") {
      bPrime.insert(op2.value);
      aPrime.retain(op2.value.length);
      j++;
      continue;
    }

    // 情况3：双方都保留，取共同的最小保留长度
    if (op1 && op1.type === "retain" && op2 && op2.type === "retain") {
      const minCount = Math.min(op1.count, op2.count);
      aPrime.retain(minCount);
      bPrime.retain(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    // 情况4：A 删除，B 保留 → A 继续删除，B 跳过被删除的部分
    if (op1 && op1.type === "delete" && op2 && op2.type === "retain") {
      const minCount = Math.min(op1.count, op2.count);
      aPrime.delete(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    // 情况5：A 保留，B 删除 → B 继续删除，A 跳过被删除的部分
    if (op1 && op1.type === "retain" && op2 && op2.type === "delete") {
      const minCount = Math.min(op1.count, op2.count);
      bPrime.delete(minCount);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    // 情况6：双方都删除，取公共最小值，双方各自消耗
    if (op1 && op1.type === "delete" && op2 && op2.type === "delete") {
      const minCount = Math.min(op1.count, op2.count);
      op1.count -= minCount;
      op2.count -= minCount;
      if (op1.count === 0) i++;
      if (op2.count === 0) j++;
      continue;
    }

    // 情况7：A 插入，B 删除 → 让 B 的删除操作跳过 A 插入的字符
    if (op1 && op1.type === "insert" && op2 && op2.type === "delete") {
      aPrime.insert(op1.value);
      bPrime.retain(op1.value.length);
      i++;
      continue;
    }

    // 情况8：A 删除，B 插入 → 让 A 的删除操作跳过 B 插入的字符
    if (op1 && op1.type === "delete" && op2 && op2.type === "insert") {
      bPrime.insert(op2.value);
      aPrime.retain(op2.value.length);
      j++;
      continue;
    }

    throw new Error("未知的操作组合");
  }

  return [aPrime, bPrime]; // 返回转换后的操作描述对象
}

module.exports = {
  transform,
};
```

### TextOperation.js

TextOperation 操作类与上一节课一致，包含 retain、insert、delete、apply、invert、toJSON 方法。

```js
class TextOperation {
  constructor() {
    this.ops = [];
  }

  retain(n) {
    if (n > 0) {
      this.ops.push({ type: "retain", count: n });
    }
    return this;
  }

  insert(str) {
    if (str && str.length > 0) {
      this.ops.push({ type: "insert", value: str });
    }
    return this;
  }

  delete(n) {
    if (n > 0) {
      this.ops.push({ type: "delete", count: n });
    }
    return this;
  }

  apply(text) {
    let result = "";
    let index = 0;
    for (const op of this.ops) {
      if (op.type === "retain") {
        result += text.slice(index, index + op.count);
        index += op.count;
      } else if (op.type === "insert") {
        result += op.value;
      } else if (op.type === "delete") {
        index += op.count;
      }
    }
    result += text.slice(index);
    return result;
  }

  invert(baseText) {
    const inverse = new TextOperation();
    let index = 0;
    for (const op of this.ops) {
      if (op.type === "retain") {
        inverse.retain(op.count);
        index += op.count;
      } else if (op.type === "insert") {
        inverse.delete(op.value.length);
      } else if (op.type === "delete") {
        const deletedText = baseText.slice(index, index + op.count);
        inverse.insert(deletedText);
        index += op.count;
      }
    }
    return inverse;
  }

  toJSON() {
    return this.ops;
  }
}

module.exports = {
  TextOperation,
};
```

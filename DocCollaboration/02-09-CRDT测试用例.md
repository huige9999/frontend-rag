# CRDT测试用例

> 本节课主要通过完整的测试用例来验证 CRDT 文本类的正确性，演示两个客户端之间插入和删除操作的同步过程。

## 关键代码

### index.js

入口文件，演示了两个客户端 A 和 B 之间的协同编辑流程：客户端 A 先插入字符，同步给 B；然后 B 插入字符，同步给 A；最后 A 删除字符，同步给 B。

```js
const { Client } = require("./client");

// 创建两个客户端
const A = new Client("A");
const B = new Client("B");

// 客户端A插入一些字符
const a1 = A.insert(0, "a");
const a2 = A.insert(1, "b");
const a3 = A.insert(2, "c");
console.log("------------------");

// 接下来将上面的操作同步到客户端B
B.receive({ type: "insert", char: a1 });
B.receive({ type: "insert", char: a2 });
B.receive({ type: "insert", char: a3 });
console.log("------------------");

// 客户端B插入一个 X 在最前面
const b1 = B.insert(0, "X");
// 接下来将这个操作同步给客户端A
A.receive({ type: "insert", char: b1 });
console.log("------------------");

// 接下来客户端A删除字符b
const a4 = A.delete(2);
// 接下来将这个操作同步给客户端B
B.receive({ type: "delete", id: a4.id });
```

### client/index.js

Client 类封装了 CRDT 文本的操作，提供 insert（插入）、delete（删除）和 receive（接收远程操作）方法，每个操作都会记录到历史记录中。

```js
const { CrdtText } = require("../crdt/crdttext.js");

class Client {
  /**
   *
   * @param {*} id 客户端id
   */
  constructor(id) {
    this.id = id; // 当前客户端id
    this.text = new CrdtText(id); // 创建一个新的 CrdtText 实例对象
    this.history = []; // 用于记录已经执行的本地操作
  }

  /**
   * 返回当前的文本内容
   */
  value() {
    return this.text.value(); // 返回当前的文本内容，不包含被删除的字符
  }

  /**
   * 插入字符
   * @param {*} index 要插入的字符位置
   * @param {*} value 要插入的字符
   */
  insert(index, value) {
    const op = this.text.localInsert(index, value); // 插入操作完成拿到字符对象
    // 接下来需要做一个历史记录
    this.history.push({
      type: "insert",
      char: op,
    });
    console.log(
      `[${this.id}] 插入 '${value}' at ${index} => ${this.text.value()}`
    );
    return op; // 返回当前的字符对象
  }

  /**
   * 删除字符方法
   * @param {*} index 要删除的字符的位置
   */
  delete(index) {
    const op = this.text.localDelete(index); // 删除操作完成，拿到删除的字符对象
    // 接下来需要做一个历史记录
    this.history.push({
      type: "delete",
      id: op.id,
    });
    console.log(`[${this.id}] 删除 index ${index} => ${this.text.value()}`);
    return op; // 返回当前的字符对象
  }

  /**
   * 接收远程操作
   * @param {*} handleDesc 操作描述对象
   * { type: "insert", char: op }
   * { type: "delete", id: op.id }
   */
  receive(handleDesc) {
    if (handleDesc.type === "insert") {
      // 说明要同步的操作是插入操作
      this.text.remoteInsert(handleDesc.char); // 远程插入
    } else if (handleDesc.type === "delete") {
      // 说明要同步的操作是删除操作
      this.text.remoteDelete(handleDesc.id); // 远程删除
    } else {
      throw new Error("操作类型不正确");
    }
    console.log(`[${this.id}] 同步操作 => ${this.text.value()}`);
  }
}

module.exports = {
  Client,
};
```

### crdt/crdttext.js

CRDT 文本类核心实现，包含 CrdtChar（字符节点）和 CrdtText（文本管理）两个类。每个字符拥有全局唯一 ID（由 counter 和 clientId 组成），插入时按 ID 排序，删除采用软删除标记。

```js
class CrdtChar {
  constructor(id, value, deleted = false) {
    this.id = id; // 唯一id
    this.value = value; // 字符内容
    this.deleted = deleted; // 是否被删除
  }
}

class CrdtText {
  constructor(clientId) {
    this.clientId = clientId; // 客户端id
    this.chars = []; // 存储所有的字符串节点
    this.counter = 0; // 本地的自增计数器
  }

  // 生成全局的唯一id [counter, clientId]
  _genId() {
    return [this.counter++, this.clientId];
  }

  // 比较两个id的大小
  static compare(id1, id2) {
    if (id1[0] !== id2[0]) return id1[0] - id2[0];
    return id2[1].localeCompare(id1[1]);
  }

  // 判断两个字符的 id 是否完全相同
  static equalId(id1, id2) {
    return id1[0] === id2[0] && id1[1] === id2[1];
  }

  // 返回当前文本中所有未删除字符组成的字符串
  value() {
    return this.chars
      .filter((c) => !c.deleted)
      .map((c) => c.value)
      .join("");
  }

  // 按 ID 排序插入字符
  _insertCharSoted(char) {
    const idx = this.chars.findIndex(
      (c) => CrdtText.compare(char.id, c.id) < 0
    );
    if (idx === -1) {
      this.chars.push(char);
    } else {
      this.chars.splice(idx, 0, char);
    }
  }

  // 本地插入
  localInsert(index, value) {
    const id = this._genId();
    const char = new CrdtChar(id, value);
    this._insertCharSoted(char);
    return char;
  }

  // 远程插入
  remoteInsert(remoteChar) {
    this._insertCharSoted(remoteChar);
  }

  // 本地删除（软删除）
  localDelete(index) {
    const char = this.chars[index];
    if (char) char.deleted = true;
    return char;
  }

  // 远程删除（软删除）
  remoteDelete(id) {
    const char = this.chars.find((c) => CrdtText.equalId(c.id, id));
    if (char) char.deleted = true;
  }
}

module.exports = {
  CrdtChar,
  CrdtText,
};
```

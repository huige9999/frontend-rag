// 数据转换函数：将嵌套配置对象转为 treemap 数据格式
function convert(source) {
  return (function inner(source, target, basePath) {
    for (var key in source) {
      var path = basePath ? (basePath + '.' + key) : key;
      if (key.indexOf('$') == -1) {
        target.children = target.children || [];
        var child = { name: path };
        target.children.push(child);
        inner(source[key], child, path);
      }
    }
    if (!target.children) {
      target.value = source.$count || 1;
    } else {
      if (source.$count != undefined) {
        target.children.push({
          name: basePath,
          value: source.$count,
        });
      }
    }
    return target;
  })(source, [], '');
}

### 代码说明
- 本课源码与第 38 课相同，可结合第 38 课的详细说明一起学习
- 重点理解 `levels` 多层级配置的递进关系：顶层颜色按索引映射，子层级通过 `colorSaturation` 调整饱和度
- `breadcrumb` 面包屑导航使用户在下钻后可以方便地返回上层
- `convert()` 函数是数据预处理的关键，将 ECharts 配置项统计原始数据转为可视化所需的树形结构

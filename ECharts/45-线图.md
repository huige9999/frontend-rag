# 45-线图

## 简介
本节课学习 ECharts 线图（lines）系列的使用，结合地理坐标系实现城市间迁徙路线的可视化，包含尾迹特效、飞机符号和涟漪气泡等效果。

## 关键代码

```javascript
// 城市坐标
var geoCoordMap = {
    '上海': [121.4648, 31.2891],
    '北京': [116.4551, 40.2539],
    // ... 更多城市坐标
};

// 北京出发 Top10 数据
var BJData = [
    [{ name: '北京' }, { name: '上海', value: 95 }],
    [{ name: '北京' }, { name: '广州', value: 90 }],
    // ... 更多数据
];

// 上海出发 Top10 数据
var SHData = [
    [{ name: '上海' }, { name: '包头', value: 95 }],
    // ... 更多数据
];

// 广州出发 Top10 数据
var GZData = [
    [{ name: '广州' }, { name: '福州', value: 95 }],
    // ... 更多数据
];

// 飞机图形 SVG 路径
var planePath = 'path://M1705.06,1318.313v-89.254l-319.9-221.799...';

// 数据转换函数：将起点终点转换为坐标数组
function convertData(data) {
    var res = [];
    for (var i = 0; i < data.length; i++) {
        var dataItem = data[i];
        var fromCoord = geoCoordMap[dataItem[0].name];
        var toCoord = geoCoordMap[dataItem[1].name];
        res.push({
            fromName: dataItem[0].name,
            toName: dataItem[1].name,
            coords: [fromCoord, toCoord]
        });
    }
    return res;
}

// 动态构建 series 数组
var series = [];
var bsg = [
    ['北京', BJData, '#a6c84c'],
    ['上海', SHData, '#ffa022'],
    ['广州', GZData, '#46bee9'],
];
bsg.forEach(function (item, i) {
    series.push(
        {
            // 尾迹特效线
            name: item[0] + ' Top10',
            type: 'lines',
            zlevel: 1,
            effect: {
                show: true,
                period: 6,
                symbolSize: 3,
                color: '#fff',
                trailLength: 0.7,
                loop: true,
            },
            lineStyle: {
                width: 0,
                color: item[2],
                curveness: 0.2,
            },
            data: convertData(item[1])
        },
        {
            // 飞机符号线
            name: item[0] + ' Top10',
            type: 'lines',
            zlevel: 2,
            symbol: ['none', 'arrow'],
            effect: {
                show: true,
                period: 6,
                trailLength: 0,
                symbol: planePath,
                symbolSize: 15,
            },
            lineStyle: {
                width: 1,
                color: item[2],
                opacity: 0.6,
                curveness: 0.2,
            },
            data: convertData(item[1])
        },
        {
            // 涟漪气泡（终点城市）
            name: item[0] + ' Top10',
            type: 'effectScatter',
            coordinateSystem: 'geo',
            zlevel: 2,
            rippleEffect: {
                brushType: 'stroke',
            },
            itemStyle: { color: item[2] },
            symbolSize: function (val) {
                return val[2] / 5;
            },
            label: {
                show: true,
                formatter: '{b}',
                position: 'right',
            },
            data: item[1].map(function (dataItem) {
                return {
                    name: dataItem[1].name,
                    value: geoCoordMap[dataItem[1].name].concat([dataItem[1].value])
                };
            }),
        }
    );
});

option = {
    backgroundColor: '#404a59',
    title: {
        text: '模拟迁徙',
        subtext: '数据纯属虚构',
        left: 'center',
        textStyle: { color: '#fff' }
    },
    legend: {
        data: ['北京 Top10', '上海 Top10', '广州 Top10'],
        textStyle: { color: '#fff' },
        left: 'right',
        top: 'bottom',
        orient: 'vertical',
        selectedMode: 'single',
    },
    geo: {
        map: 'china',
        roam: true,
        itemStyle: {
            areaColor: '#323c48',
            borderColor: '#404a59'
        },
        emphasis: {
            itemStyle: { areaColor: '#2a333d' }
        }
    },
    series: series,
};
```

### 代码说明
- `type: 'lines'` - 线图系列类型，用于绘制起点到终点的线段，数据中需提供 `coords: [[起点经纬度], [终点经纬度]]`
- `effect` - 线特效配置
  - `trailLength: 0.7` - 尾迹长度，0表示无尾迹，1表示完整尾迹
  - `symbol: planePath` - 使用自定义 SVG path 作为流动的图标（飞机形状）
  - `period` - 动画周期（秒）
- `symbol: ['none', 'arrow']` - 线的起点和终点图标，起点无图标，终点为箭头
- `curveness: 0.2` - 线的弯曲度，使线条呈现弧形
- `type: 'effectScatter'` - 涟漪散点图，用于在终点城市显示带涟漪效果的气泡
- `selectedMode: 'single'` - 图例选择模式为单选，一次只显示一个城市的迁徙路线
- `convertData()` - 数据转换函数，将起终点城市名转为经纬度坐标对

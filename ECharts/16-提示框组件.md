# 16-提示框组件

## 简介
本节课学习 tooltip（提示框）组件的详细配置，包括触发方式、位置控制、内容格式化、样式自定义等核心功能，结合多 Y 轴柱状图+折线图进行演示。

## 关键代码

### tooltip 提示框完整配置

```javascript
var myChart = echarts.init(document.getElementById('chart'));
var colors = ['#5793f3', '#d14a61', '#675bba'];

myChart.setOption({
    color: colors,
    legend: {},
    grid: {
        right: "20%",
    },
    tooltip: {
        // axisPointer: {
        //     type: 'cross',
        // }
        show: true,                     // 是否显示提示框
        trigger: 'axis',                // 触发类型：item / axis / none
        showContent: true,              // 是否显示提示框浮层
        alwaysShowContent: false,       // 是否总是显示提示框
        triggerOn: 'mousemove',         // 触发条件：mousemove / click / mousemove|click
        // showDelay: 500,              // 显示延迟（毫秒）
        hideDelay: 1000,                // 隐藏延迟（毫秒）
        // enterable: true,             // 鼠标是否可进入提示框浮层中
        confine: false,                 // 是否将 tooltip 限制在图表区域内
        transitionDuration: 0.4,        // 提示框移动动画时间

        // 位置设置（三种方式）
        // position: [10, 10],                          // 绝对像素位置
        // position: ['50%', '50%'],                    // 相对百分比位置
        position: function(point, params, dom, rect, size) {
            var middle = size.viewSize[0] / 2;  // echarts容器宽度的一半
            if (point[0] < middle) {
                // 鼠标在容器左半边
                return {
                    left: point[0],
                    top: point[1],
                };
            } else {
                // 鼠标在容器右半边
                return {
                    left: point[0] - size.contentSize[0],
                    top: point[1],
                };
            }
        },

        // 格式化内容（字符串模板方式）
        // formatter: '{b}份的数据：<br> {a0} : {c0} <br> {a1} : {c1} <br> {a2} : {c2} <br>',

        // 格式化内容（函数方式）
        formatter: function(params, ticket, callback) {
            var str = params[0].name + '份的数据：<br>'
                + params[0].seriesName + ' : ' + params[0].value + '<br>'
                + params[1].seriesName + ' : ' + params[1].value + '<br>'
                + params[2].seriesName + ' : ' + params[2].value;
            return str;
        },

        backgroundColor: 'rgba(255,50,50,0.7)',   // 背景色（支持透明度）
        borderColor: 'green',                      // 边框颜色
        borderWidth: 2,                            // 边框宽度
        textStyle: {                               // 文本样式
            color: '#000',
        },
        extraCssText: 'box-shadow: 0 0 3px rgba(0, 0, 0, 1);',  // 额外CSS样式
    },
    xAxis: {
        axisTick: { alignWithLabel: true },
        data: ['1月', '2月', '3月', '4月', '5月', '6月',
               '7月', '8月', '9月', '10月', '11月', '12月'],
    },
    yAxis: [
        {
            name: '蒸发量',
            position: 'right',
            min: 0, max: 250,
            axisLabel: { formatter: '{value} ml' },
            axisLine: { lineStyle: { color: colors[0] } }
        },
        {
            name: '降水量',
            position: 'right',
            offset: 80,
            min: 0, max: 250,
            axisLabel: { formatter: '{value} ml' },
            axisLine: { lineStyle: { color: colors[1] } }
        },
        {
            name: '温度',
            min: 0, max: 25,
            axisLabel: { formatter: '{value} °C' },
            axisLine: { lineStyle: { color: colors[2] } }
        },
    ],
    series: [
        {
            name: '蒸发量', type: 'bar', yAxisIndex: 0,
            data: [2.0, 4.9, 7.0, 23.2, 25.6, 76.7, 135.6, 162.2, 32.6, 20.0, 6.4, 3.3]
        },
        {
            name: '降水量', type: 'bar', yAxisIndex: 1,
            data: [2.6, 5.9, 9.0, 26.4, 28.7, 70.7, 175.6, 182.2, 48.7, 18.8, 6.0, 2.3]
        },
        {
            name: '平均温度', type: 'line', yAxisIndex: 2,
            data: [2.0, 2.2, 3.3, 4.5, 6.3, 10.2, 20.3, 23.4, 23.0, 16.5, 12.0, 6.2]
        },
    ]
});
```

### 代码说明
- `trigger: 'axis'` - 触发类型，'item' 用于散点图等非轴类图表，'axis' 用于柱状图/折线图等轴类图表
- `triggerOn` - 触发条件，支持 'mousemove'（鼠标移动）、'click'（点击）及其组合
- `showDelay / hideDelay` - 显示和隐藏的延迟时间（毫秒）
- `alwaysShowContent` - 是否总是显示提示框内容，不随鼠标移出而隐藏
- `enterable` - 是否允许鼠标进入提示框浮层，适用于需要与提示框交互的场景
- `confine` - 是否将 tooltip 限制在图表的 DOM 容器内
- `position` - 提示框位置，支持三种写法：像素数组 `[x, y]`、百分比数组 `['50%', '50%']`、函数回调
- `position` 函数参数：`point`（鼠标位置）、`params`（数据）、`dom`（DOM元素）、`rect`（图形包围盒）、`size`（包含 viewSize 和 contentSize）
- `formatter` - 内容格式化，支持字符串模板（`{a}` 系列名、`{b}` 数据名、`{c}` 数据值）和函数两种方式
- `backgroundColor` - 支持纯色和 rgba 透明色
- `extraCssText` - 额外的 CSS 样式字符串，用于自定义提示框的外观

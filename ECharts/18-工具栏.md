# 18-工具栏

## 简介
本节课学习 toolbox（工具栏）组件的配置，包括 saveAsImage（保存为图片）、restore（还原）、dataView（数据视图）、dataZoom（数据区域缩放）、magicType（动态类型切换）等内置工具的功能。

## 关键代码

### toolbox 工具栏完整配置

```javascript
var myChart = echarts.init(document.getElementById('chart'));
myChart.setOption({
    title: {
        text: '未来一周气温变化',
        subtext: '纯属虚构',
    },
    legend: {},
    tooltip: {
        trigger: 'axis',
    },
    toolbox: {
        right: '10%',             // 工具栏位置
        show: true,               // 是否显示工具栏
        itemSize: 20,             // 工具图标大小
        showTitle: true,          // 是否显示每个工具的标题
        feature: {
            // ===== 保存为图片 =====
            saveAsImage: {
                type: 'gif',       // 保存的图片类型
                name: 'kaivon',    // 保存的文件名
                backgroundColor: 'pink',            // 保存图片的背景色
                excludeComponents: ['toolbox', 'title', 'legend'],  // 排除的组件
                show: true,
                title: '保存一下',
                iconStyle: {                        // 图标样式
                    color: 'green',
                    borderColor: 'red',
                    borderWidth: 1,
                },
                emphasis: {                          // hover 时的样式
                    iconStyle: {
                        color: 'red',
                        borderColor: 'green',
                        borderWidth: 1,
                    }
                },
                pixelRatio: 2,    // 分辨率比例
            },

            // ===== 还原配置 =====
            restore: {
                show: true,
                title: '恢复',
            },

            // ===== 数据视图 =====
            dataView: {
                show: true,
                readOnly: false,    // 是否只读（false 可编辑）
                // 自定义数据视图的内容渲染
                optionToContent: function(opt) {
                    var axisData = opt.xAxis[0].data;
                    var series = opt.series;
                    var table = '<table style="width:100%;text-align:center"><tbody><tr>'
                        + '<td>时间</td>'
                        + '<td>' + series[0].name + '</td>'
                        + '<td>' + series[1].name + '</td>'
                        + '</tr>';
                    for (var i = 0, l = axisData.length; i < l; i++) {
                        table += '<tr>'
                            + '<td>' + axisData[i] + '</td>'
                            + '<td>' + series[0].data[i] + '</td>'
                            + '<td>' + series[1].data[i] + '</td>'
                            + '</tr>';
                    }
                    table += '</tbody></table>';
                    return table;
                },
                // 用户点击刷新按钮时触发
                contentToOption: function(container, opt) {
                    // 可以在这里处理编辑后的数据
                },
                lang: ['数据视图1', '关闭1', '刷新1'],   // 自定义按钮文本
                backgroundColor: 'gray',                  // 数据视图背景色
            },

            // ===== 数据区域缩放 =====
            dataZoom: {
                show: true,
                title: {
                    zoom: '缩放',
                    back: '还原',
                },
                xAxisIndex: 'true',    // 指定控制哪些 x 轴
            },

            // ===== 动态类型切换 =====
            magicType: {
                show: true,
                type: ['line', 'bar', 'stack', 'tiled'],  // 可切换的图表类型
                title: {
                    line: '换成线',
                },
                option: {
                    line: {
                        // 切换到 line 类型时的额外配置
                    }
                }
            },

            // ===== 区域选择（刷选）=====
            brush: {
                // 空配置，使用默认值
            }
        }
    },
    xAxis: {
        boundaryGap: false,
        data: ['周一', '周二', '周三', '周四', '周五', '周六', '周日'],
    },
    yAxis: {
        axisLabel: { formatter: '{value} °C' },
    },
    series: [
        {
            name: '最高气温',
            type: 'line',
            data: [11, 11, 15, 13, 12, 13, 10],
        },
        {
            name: '最低气温',
            type: 'line',
            data: [1, -2, 2, 5, 3, 2, 0],
        },
    ]
});
```

### 代码说明
- `toolbox` - 工具栏组件，提供一组内置工具方便用户操作图表
- `itemSize` - 工具栏图标的大小
- `showTitle` - 是否在 hover 时显示工具标题提示
- `saveAsImage` - 保存图表为图片，支持指定类型、文件名、背景色、排除组件等
- `excludeComponents` - 保存图片时排除不需要的组件（如工具栏、标题、图例）
- `restore` - 还原图表到初始状态（取消所有缩放、类型切换等操作）
- `dataView` - 数据视图工具，以表格形式展示原始数据，支持编辑（`readOnly: false`）
- `optionToContent` - 自定义数据视图的渲染内容，接收 option 参数，返回 HTML 字符串
- `contentToOption` - 从数据视图的内容还原回 option（用户编辑数据后点刷新时触发）
- `dataZoom` - 工具栏中的数据缩放按钮，用户可以框选区域进行缩放
- `magicType` - 动态类型切换，支持在 line（折线）、bar（柱状）、stack（堆叠）、tiled（平铺）之间切换
- `brush` - 区域选择工具，用于在图表上框选数据区域

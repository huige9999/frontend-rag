# 44-平行坐标图

## 简介
本节课学习 ECharts 平行坐标图（parallel）系列的使用，通过结合世界地图散点图和平行坐标系，实现多维数据的可视化与联动分析。

## 关键代码

```javascript
// 城市坐标
var geoCoordMap = {
    "Amsterdam": [4.895168, 52.370216],
    "Beijing": [116.407395, 39.904211],
    // ... 更多城市坐标
};

// 统计维度定义
var schema = [
    "Cities",
    "Gross purchasing power",
    "Net purchasing power",
    "Prices (excl. rent)",
    // ... 更多维度
];

// 原始数据（每行对应一个城市的各项指标）
var rawData = [
    ["Amsterdam", 101.6, 90.1, 77.1, ...],
    ["Beijing", 28.2, 29.9, 60.3, ...],
    // ... 更多城市数据
];

// 数据转换函数：将城市名与经纬度和指标值拼接
function converData(data) {
    var res = [];
    for (var i = 0; i < data.length; i++) {
        var geoData = geoCoordMap[data[i][0]];
        res.push({
            name: data[i][0],
            value: geoData.concat(data[i].slice(1)),
        });
    }
    return res;
}

// 生成平行坐标系的坐标轴配置
function makeParallelAxis(schema) {
    var parallelAxis = [];
    for (var i = 1; i < schema.length; i++) {
        parallelAxis.push({ dim: i, name: schema[i], realtime: false });
    }
    return parallelAxis;
}

option = {
    backgroundColor: '#404a59',
    title: {
        text: 'Prices and Earnings 2012',
        subtext: 'data from macrofocus',
        left: 'center',
        top: 5,
        textStyle: { color: '#eee' },
    },
    brush: {
        geoIndex: 0,
        brushLink: 'all',
        inBrush: {
            opacity: 1,
            symbolSize: 20,
        },
        outOfBrush: {
            color: '#000',
            opacity: 0.2,
        }
    },
    geo: {
        map: 'world',
        left: '6%',
        right: '14%',
        top: 40,
        bottom: '54%',
        roam: true,
        itemStyle: {
            borderWidth: 0.2,
            borderColor: '#111',
            areaColor: '#323c48'
        },
    },
    parallel: {
        top: '50%',
        left: 60,
        right: 20,
        bottom: 100,
        axisExpandable: true,
        axisExpandCenter: 15,
        axisExpandCount: 10,
        axisExpandWidth: 60,
        axisExpandTriggerOn: 'mousemove',
        parallelAxisDefault: {
            nameLocation: 'start',
            nameRotate: 25,
            nameGap: 20,
            axisLine: {
                lineStyle: {
                    color: 'rgba(255,255,255,0.3)'
                }
            },
            axisTick: { show: false },
        }
    },
    parallelAxis: makeParallelAxis(schema),
    series: [
        {
            // 地图上的散点
            name: 'Prices and Earnings 2012',
            type: 'scatter',
            coordinateSystem: 'geo',
            data: converData(rawData),
            symbolSize: 10,
            itemStyle: {
                color: '#577ceb',
                borderColor: '#fff',
            },
            emphasis: {
                label: {
                    show: true,
                    formatter: '{b}',
                    position: 'right',
                }
            }
        },
        {
            // 平行坐标图表
            name: 'parallel',
            type: 'parallel',
            data: rawData,
            lineStyle: {
                color: '#577ceb',
                width: 1,
            },
            activeOpacity: 1,
            inactiveOpacity: 0,
            smooth: true,
        }
    ],
    blendMode: 'lighter',
};
```

### 代码说明
- `type: 'parallel'` - 指定系列类型为平行坐标图
- `parallel` - 平行坐标系的全局配置组件
  - `axisExpandable: true` - 允许坐标轴展开，解决维度过多时的显示问题
  - `axisExpandCenter` / `axisExpandCount` - 控制展开的中心位置和展开数量
  - `parallelAxisDefault` - 各坐标轴的默认样式配置
- `parallelAxis` - 通过 `makeParallelAxis()` 动态生成各维度的坐标轴配置，`dim` 指定对应数据维度
- `brush` - 区域选择组件，实现地图散点与平行坐标图之间的联动高亮
  - `brushLink: 'all'` - 所有 series 之间联动
  - `inBrush` / `outOfBrush` - 选中/未选中区域的样式
- `geo` - 地理坐标系配置，使用世界地图（需引入 word.js）
- `smooth: true` - 平行坐标线使用平滑曲线
- `blendMode: 'lighter'` - 混合模式，使重叠部分更亮

# 15-视觉映射组件-3

## 简介
本节课学习 visualMap 的 piecewise（分段型）视觉映射组件，包括使用 splitNumber 自动分段和 pieces 自定义分段两种方式，并结合地图与散点图展示 category 类型的分段映射。

## 关键代码

### 分段型视觉映射 - 地图示例

```javascript
var myChart1 = echarts.init(document.getElementById('chart1'));
myChart1.setOption({
    visualMap: [
        {
            type: 'piecewise',       // 分段型视觉映射
            splitNumber: 5,          // 自动分段数量
            min: 0,
            max: 2000,

            // 自定义分段（注释状态，可替代 splitNumber）
            // pieces: [
            //     { min: 1500 },                          // 不指定 max，表示 max 为无限大
            //     { min: 900, max: 1500 },
            //     { min: 310, max: 1000 },
            //     { min: 200, max: 300 },
            //     { min: 10, max: 200, label: '10 到 200（自定义label）' },
            //     { value: 123, label: '123（自定义特殊颜色）', color: 'grey' },
            //     { max: 5 }                               // 不指定 min，表示 min 为无限小
            // ],

            // 使用 gt/lt/lte 等操作符定义分段
            // pieces: [
            //     { gt: 1500 },            // (1500, Infinity]
            //     { gt: 900, lte: 1500 },  // (900, 1500]
            //     { gt: 310, lte: 1000 },  // (310, 1000]
            //     { gt: 200, lte: 300 },   // (200, 300]
            //     { gt: 10, lte: 200, label: '10 到 200（自定义label）' },
            //     { value: 123, label: '123（自定义特殊颜色）', color: 'grey' },
            //     { lt: 5 }                // (-Infinity, 5)
            // ],

            minOpen: true,                // 是否展示大于 max 的项
            maxOpen: true,                // 是否展示小于 min 的项
            selectedMode: 'multiple',     // 选中模式：multiple / single
            inverse: true,                // 是否反转
            precision: 1,                 // 数据精度
            itemWidth: 30,                // 每项图形宽度
            itemHeight: 20,               // 每项图形高度
            align: 'left',                // 对齐方式：auto / left / right
            text: ['high', 'low'],        // 两端文本
            showLabel: true,              // 是否显示每项的文本标签
            textGap: 20,                  // 文本与图例间距
            itemGap: 30,                  // 各项之间的间距
            orient: 'horizontal'          // 方向
        }
    ],
    series: [{
        type: 'map',
        mapType: 'china',
        roam: true,
        data: [
            { name: '北京', value: random() },
            { name: '天津', value: random() },
            // ... 其他省份数据
        ]
    }]
});
```

### 分段型视觉映射 - 散点图示例（categories 分类映射）

```javascript
var myChart2 = echarts.init(document.getElementById('chart2'));
myChart2.setOption({
    legend: {},
    xAxis: {
        name: '日期',
        nameTextStyle: { fontSize: 14 },
        splitLine: { show: false }
    },
    yAxis: {
        name: 'AQI指数',
        nameTextStyle: { fontSize: 20 },
        splitLine: { show: false }
    },
    visualMap: [
        {
            type: 'piecewise',
            right: 0,
            bottom: 80,
            dimension: 7,                    // 指定映射到数据的第7个维度
            categories: [                    // 分类数据
                '严重污染', '重度污染', '中度污染',
                '轻度污染', '良', '优'
            ],
            inRange: {
                symbolSize: 30,              // 范围内的统一符号大小
                symbol: {
                    '优': 'diamond',         // 指定分类的符号形状
                    '良': 'pin',
                }
            },
            outOfRange: {
                symbolSize: {
                    '优': 50,                // 超出范围时的符号大小
                    '': 30,
                },
                symbol: {
                    '优': 'triangle',        // 超出范围时的符号形状
                    '良': 'arrow',
                }
            },
        }
    ],
    series: [
        {
            name: '北京',
            type: 'scatter',
            data: dataBJ,    // [日期, AQI, ... , 等级]
        },
        {
            name: '上海',
            type: 'scatter',
            data: dataSH,
        },
        {
            name: '广州',
            type: 'scatter',
            data: dataGZ,
        }
    ]
});
```

### 代码说明
- `type: 'piecewise'` - 分段型视觉映射，将数据分成离散的段进行映射
- `splitNumber` - 自动分段的段数，配合 min/max 使用
- `pieces` - 自定义分段，每段可指定 min/max/gt/lt/lte 等条件，以及 label 和 color
- `minOpen / maxOpen` - 是否在范围之外额外展示一个"大于 max"或"小于 min"的项
- `selectedMode` - 选中模式，'multiple' 多选，'single' 单选
- `categories` - 使用分类数据而非数值区间进行分段映射
- `dimension` - 指定数据中哪个维度参与视觉映射（散点图数据是多维数组）
- `inRange.symbol` - 可以为不同分类指定不同的图形符号
- `outOfRange` - 超出映射范围的视觉样式定义

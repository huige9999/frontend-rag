# 1-7 文件流

## 知识要点

### 为什么需要文件流？

使用 `fs.readFile` 读取大文件时会一次性将整个文件加载到内存，可能导致内存溢出。**文件流（Stream）** 通过分块处理数据解决此问题。

### 流的类型

| 类型 | 说明 |
|------|------|
| 可读流（Readable） | `fs.createReadStream(path)` |
| 可写流（Writable） | `fs.createWriteStream(path)` |

### 核心概念

- **highWaterMark** — 每次读取/写入的最大字节数（水位线）
- **背压（Backpressure）** — 写入速度慢于读取速度时，需要暂停读取
- **pipe** — 管道方法，自动处理背压

### 可读流事件

- `open` — 文件打开
- `data` — 接收到数据块
- `pause` / `resume` — 暂停/恢复
- `end` — 数据读取完毕
- `close` — 流关闭
- `error` — 出错

### 可写流事件

- `drain` — 可以继续写入（缓冲区已清空）

## 代码示例

### 1. 可读流（readStream.js）

```js
const fs = require("fs");
const path = require("path");

const filename = path.resolve(__dirname, "./abc.txt");
const rs = fs.createReadStream(filename, {
  encoding: "utf-8",
  highWaterMark: 1,      // 每次读取1个字节
  autoClose: true         // 读完后自动关闭
});

rs.on("open", () => {
  console.log("文件被打开了");
});

rs.on("error", () => {
  console.log("出错了！！");
});

rs.on("close", () => {
  console.log("文件关闭了");
});

rs.on("data", chunk => {
  console.log("读到了一部分数据：", chunk);
  rs.pause(); // 暂停读取
});

rs.on("pause", () => {
  console.log("暂停了");
  setTimeout(() => {
    rs.resume(); // 1秒后恢复
  }, 1000);
});

rs.on("resume", () => {
  console.log("恢复了");
});

rs.on("end", () => {
  console.log("全部数据读取完毕");
});
```

### 2. 可写流（write.js）

```js
const fs = require("fs");
const path = require("path");

const filename = path.resolve(__dirname, "./temp/abc.txt");
const ws = fs.createWriteStream(filename, {
  encoding: "utf-8",
  highWaterMark: 16 * 1024  // 16KB
});

let i = 0;
// 一直写，直到到达上限或无法再直接写入
function write() {
  let flag = true;
  while (i < 1024 * 1024 * 10 && flag) {
    flag = ws.write("a"); // 返回值表示是否还能继续写入
    i++;
  }
}

write();

ws.on("drain", () => {
  write(); // 缓冲区清空后继续写入
});
```

### 3. 流式文件复制 — 手动处理背压（copy.js）

```js
const fs = require("fs");
const path = require("path");

async function method2() {
  const from = path.resolve(__dirname, "./temp/abc.txt");
  const to = path.resolve(__dirname, "./temp/abc2.txt");
  console.time("方式2");

  const rs = fs.createReadStream(from);
  const ws = fs.createWriteStream(to);

  rs.on("data", chunk => {
    const flag = ws.write(chunk);
    if (!flag) {
      // 下一次写入会造成背压，暂停读取
      rs.pause();
    }
  });

  ws.on("drain", () => {
    // 可以继续写了
    rs.resume();
  });

  rs.on("close", () => {
    ws.end(); // 完毕写入流
    console.timeEnd("方式2");
    console.log("复制完成");
  });
}
method2();
```

### 4. 流式文件复制 — 使用 pipe（index.js）

`pipe()` 方法自动处理背压，是推荐的流复制方式：

```js
const fs = require("fs");
const path = require("path");

async function method2() {
  const from = path.resolve(__dirname, "./temp/abc.txt");
  const to = path.resolve(__dirname, "./temp/abc2.txt");
  console.time("方式2");

  const rs = fs.createReadStream(from);
  const ws = fs.createWriteStream(to);

  rs.pipe(ws); // pipe 自动处理背压

  rs.on("close", () => {
    console.timeEnd("方式2");
  });
}
method2();
```

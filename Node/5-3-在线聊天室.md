# 5-3 [扩展] 在线聊天室

## 知识要点

本节课综合运用 Socket.IO 实现一个多人在线聊天室，包含以下功能：

1. **用户登录** — 输入昵称进入聊天室，昵称不可重复
2. **在线用户列表** — 实时显示当前在线用户
3. **用户进入/退出通知** — 实时通知用户上下线
4. **私聊与群聊** — 可以选择对所有人发送消息，也可以指定用户私聊

### 消息规格

#### 客户端发送的消息

| 消息名称 | 消息内容 | 说明 |
|---------|---------|------|
| `users` | 无 | 获取当前所有在线用户 |
| `login` | 用户名（字符串） | 登录 |
| `msg` | `{to: "目标用户名/null", content: "消息内容"}` | 发送消息 |

#### 服务器发送的消息

| 消息名称 | 消息内容 | 说明 |
|---------|---------|------|
| `users` | 用户数组 | 返回在线用户列表 |
| `login` | true / false | 登录结果（false 表示昵称已存在） |
| `userin` | 用户名 | 新用户进入 |
| `userout` | 用户名 | 用户离开 |
| `new msg` | `{from, content, to}` | 新消息 |

## 关键代码

### 入口文件（index.js）

Express 创建服务器，将 Socket.IO 逻辑分离到 `chatServer.js`：

```js
const express = require("express");
const http = require("http");
const path = require("path");

const app = express();
const server = http.createServer(app);
app.use(express.static(path.resolve(__dirname, "public")));

// 将 websocket 逻辑分离到单独模块
require("./chatServer")(server);

server.listen(5008, () => {
  console.log("server listening on 5008");
});
```

### 聊天服务器核心逻辑（chatServer.js）

处理用户登录、消息转发、用户退出等核心业务：

```js
const socketIO = require("socket.io");

let users = []; // 在线用户列表

module.exports = function (server) {
  const io = socketIO(server);
  io.on("connection", (socket) => {
    let curUser = ""; // 当前用户名

    // 处理登录
    socket.on("login", (data) => {
      if (
        data === "所有人" ||
        users.filter((u) => u.username === data).length > 0
      ) {
        // 昵称不可用
        socket.emit("login", false);
      } else {
        // 昵称可用，保存用户信息
        users.push({ username: data, socket });
        curUser = data;
        socket.emit("login", true);
        // 通知其他用户有新用户进入
        socket.broadcast.emit("userin", data);
      }
    });

    // 返回在线用户列表
    socket.on("users", () => {
      const arr = users.map((u) => u.username);
      socket.emit("users", arr);
    });

    // 处理聊天消息
    socket.on("msg", (data) => {
      if (data.to) {
        // 私聊：发送给指定用户
        const us = users.filter((u) => u.username === data.to);
        const u = us[0];
        u.socket.emit("new msg", {
          from: curUser,
          content: data.content,
          to: data.to,
        });
      } else {
        // 群聊：发送给所有人（广播）
        socket.broadcast.emit("new msg", {
          from: curUser,
          content: data.content,
          to: data.to,
        });
      }
    });

    // 用户断开连接
    socket.on("disconnect", () => {
      socket.broadcast.emit("userout", curUser);
      users = users.filter((u) => u.username !== curUser);
    });
  });
};
```

### 客户端聊天逻辑（public/js/chat.js）

客户端通过 Socket.IO 与服务器通信，处理登录、消息收发：

```js
const socket = io.connect();

// 客户端发送消息给服务器
page.onLogin = function (username) {
  socket.emit("login", username);
};

page.onSendMsg = function (me, msg, to) {
  socket.emit("msg", { to, content: msg });
  page.addMsg(me, msg, to);
  page.clearInput();
};

// 客户端监听服务器消息
socket.on("login", (result) => {
  if (result) {
    page.intoChatRoom();
    socket.emit("users", ""); // 登录成功后获取在线用户
  } else {
    alert("昵称不可用，请更换昵称");
  }
});

socket.on("users", (users) => {
  page.initChatRoom();
  for (const u of users) {
    page.addUser(u);
  }
});

socket.on("userin", (username) => {
  page.addUser(username);
});

socket.on("userout", (username) => {
  page.removeUser(username);
});

socket.on("new msg", (result) => {
  page.addMsg(result.from, result.content, result.to);
});
```

### 客户端 UI 模块（public/js/ui.js）

使用 jQuery 管理页面 UI 状态（用户列表、消息列表等）：

```js
const page = (function () {
  const userList = $(".users");
  const chatList = $(".chat-list");

  function intoChatRoom() {
    $(".login").hide();
    $(".chat").show();
  }

  function addUser(userName) {
    $("<li>").text(userName).attr("user", userName).appendTo(userList);
    const number = +$(".user-list .title span").text();
    $(".user-list .title span").text(number + 1);
    addLog(`<span class="user">${userName}</span> 进入聊天室`);
    chatList.scrollTop(chatList.prop("scrollHeight"), 0);
  }

  function removeUser(userName) {
    const li = userList.find(`li[user="${userName}"]`);
    if (!li.length) return;
    li.remove();
    const number = +$(".user-list .title span").text();
    $(".user-list .title span").text(number - 1);
    addLog(`<span class="user">${userName}</span> 退出了聊天室`);
  }

  function addMsg(from, msg, to) {
    const li = $("<li>").html(
      `<span class="user">${from}</span> <span class="gray">对</span>
       <span class="user">${to ? to : "所有人"}</span>
       <span class="gray">说：</span>`
    );
    $("<span>").text(msg).appendTo(li);
    li.appendTo(chatList);
    chatList.scrollTop(chatList.prop("scrollHeight"), 0);
  }

  // 点击用户列表切换聊天对象
  userList.click((e) => {
    if (e.target.tagName === "LI") {
      $(".sendmsg .user").text(e.target.innerText);
    }
  });

  return {
    me: null, intoChatRoom, initChatRoom, addUser, addLog,
    removeUser, addMsg, clearInput, getTargetUser,
    onLogin: null, onSendMsg: null,
  };
})();

// 绑定回车事件
(function () {
  $(".login input").keydown((e) => {
    if (e.key === "Enter") {
      page.me = $(e.target).val();
      page.onLogin && page.onLogin(page.me);
    }
  });
  $(".sendmsg input").keydown((e) => {
    if (e.key === "Enter") {
      page.onSendMsg &&
        page.onSendMsg(page.me, $(e.target).val(), page.getTargetUser());
    }
  });
})();
```

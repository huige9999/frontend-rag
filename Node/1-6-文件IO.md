# 1-6 文件IO

## 知识要点

`fs`（File System）模块是 Node.js 中用于文件操作的核心模块，提供三种使用方式：

| 风格 | 示例 | 特点 |
|------|------|------|
| 回调风格 | `fs.readFile(path, callback)` | 原始方式，易产生回调地狱 |
| 同步风格 | `fs.readFileSync(path)` | 阻塞执行，影响性能 |
| Promise 风格 | `fs.promises.readFile(path)` | 推荐方式，配合 async/await |

### 常用 API

- `fs.promises.readFile(path)` — 读取文件
- `fs.promises.writeFile(path, data)` — 写入文件
- `fs.promises.stat(path)` — 获取文件/目录状态
- `fs.promises.mkdir(path)` — 创建目录
- `fs.promises.readdir(path)` — 读取目录内容

### 6. 手写 exists 函数（手写exists.js）

Node.js 已废弃 `fs.exists()`，推荐使用 `fs.stat()` 模拟实现：

```js
const fs = require("fs");
const path = require("path");
const dirname = path.resolve(__dirname, "./myfiles/3");

async function exists(filename) {
  try {
    await fs.promises.stat(filename);
    return true;
  } catch (err) {
    if (err.code === "ENOENT") {
      // 文件不存在
      return false;
    }
    throw err;
  }
}

async function test() {
  const result = await exists(dirname);
  if (result) {
    console.log("目录已存在，无需操作");
  } else {
    await fs.promises.mkdir(dirname);
    console.log("目录创建成功");
  }
}
test();
```

### 7. 手动复制文件（手动复制文件.js）

通过读取文件到内存再写入实现文件复制（适合小文件）：

```js
const fs = require("fs");
const path = require("path");

async function test() {
  const fromFilename = path.resolve(__dirname, "./myfiles/1.jpeg");
  const buffer = await fs.promises.readFile(fromFilename);
  const toFilename = path.resolve(__dirname, "./myfiles/1.copy.jpeg");
  await fs.promises.writeFile(toFilename, buffer);
  console.log("copy success！");
}
test();
```

### 8. 文件操作封装类（index.js）

将文件操作封装为 `File` 类，支持获取文件内容、子文件列表等：

```js
const fs = require("fs");
const path = require("path");

class File {
  constructor(filename, name, ext, isFile, size, createTime, updateTime) {
    this.filename = filename;
    this.name = name;
    this.ext = ext;
    this.isFile = isFile;
    this.size = size;
    this.createTime = createTime;
    this.updateTime = updateTime;
  }

  async getContent(isBuffer = false) {
    if (this.isFile) {
      if (isBuffer) {
        return await fs.promises.readFile(this.filename);
      } else {
        return await fs.promises.readFile(this.filename, "utf-8");
      }
    }
    return null;
  }

  async getChildren() {
    if (this.isFile) {
      return [];
    }
    let children = await fs.promises.readdir(this.filename);
    children = children.map(name => {
      const result = path.resolve(this.filename, name);
      return File.getFile(result);
    });
    return Promise.all(children);
  }

  static async getFile(filename) {
    const stat = await fs.promises.stat(filename);
    const name = path.basename(filename);
    const ext = path.extname(filename);
    const isFile = stat.isFile();
    const size = stat.size;
    const createTime = new Date(stat.birthtime);
    const updateTime = new Date(stat.mtime);
    return new File(filename, name, ext, isFile, size, createTime, updateTime);
  }
}

async function readDir(dirname) {
  const file = await File.getFile(dirname);
  return await file.getChildren();
}

async function test() {
  const dirname = path.resolve(__dirname, "./myfiles");
  const result = await readDir(dirname);
  const datas = await result[0].getChildren();
  console.log(datas);
}
test();
```

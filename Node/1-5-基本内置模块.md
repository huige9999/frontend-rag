# 1-5 基本内置模块

## 知识要点

Node.js 提供了许多内置模块，以下是常用的基础模块：

| 模块 | 说明 |
|------|------|
| `os` | 操作系统相关信息 |
| `path` | 文件路径处理 |
| `url` | URL 解析与格式化 |
| `util` | 实用工具函数 |

---

## 一、os 模块

提供操作系统相关信息的接口。

### 常用 API

- `os.EOL` — 系统换行符
- `os.arch()` — 系统 CPU 架构
- `os.cpus()` — CPU 信息（`os.cpus().length` 获取核心数）
- `os.freemem()` — 空闲内存（字节）
- `os.homedir()` — 用户主目录
- `os.hostname()` — 主机名
- `os.tmpdir()` — 临时文件目录

```js
const os = require("os");

console.log(os.EOL);           // 系统换行符
console.log(os.arch());        // CPU 架构，如 x64
console.log(os.cpus().length); // CPU 核心数
console.log(os.freemem() / 2 ** 30); // 空闲内存（GB）
console.log(os.homedir());     // 用户主目录
console.log(os.hostname());    // 主机名
console.log(os.tmpdir());      // 临时目录
```

## 二、path 模块

用于处理和转换文件路径。

### 常用 API

- `path.basename(path)` — 获取路径中文件名部分
- `path.sep` — 路径分隔符（`\` 或 `/`）
- `path.delimiter` — 环境变量分隔符（`;` 或 `:`）
- `path.dirname(path)` — 获取目录名
- `path.extname(path)` — 获取扩展名
- `path.join()` — 拼接路径
- `path.relative(from, to)` — 获取相对路径
- `path.resolve()` — 解析为绝对路径

```js
const path = require("path");

// 获取文件名（可去除扩展名）
const basename = path.basename("fdg/dfgdfg/adfaf/fdgdfgd/a.asf", ".html");

console.log(path.sep);  // 路径分隔符
console.log(path.dirname("a/b/c/d"));     // 'a/b/c'
console.log(path.extname("a/b/c/a.js"));  // '.js'
console.log(path.join("a/b", "../", "d.js")); // 'a/d.js'
console.log(path.relative('/data/orandea/test/aaa', '/data/orandea/impl/bbb'));

// 解析绝对路径
const absPath = path.resolve(__dirname, "./a.js");
console.log(absPath);
```

## 三、url 模块

用于 URL 的解析与格式化。

### 常用 API

- `URL.parse(urlString)` — 解析 URL 字符串为对象
- `URL.format(urlObject)` — 将 URL 对象格式化为字符串
- `url.searchParams` — URL 查询参数操作

```js
// 解析 URL
const URL = require("url");
const url = URL.parse("https://nodejs.org:80/a/b/c?t=3&u=5#abc");
console.log(url);
console.log(url.searchParams.has("t"));  // true
console.log(url.searchParams.get("t"));  // '3'
```

```js
// 格式化 URL 对象
const URL = require("url");
const obj = {
  href: "https://nodejs.org:80/a/b/c?t=3&u=5#abc",
  origin: "https://nodejs.org:80",
  protocol: "https:",
  username: "",
  password: "",
  host: "nodejs.org:80",
  hostname: "nodejs.org",
  port: "80",
  pathname: "/a/b/c",
  search: "?t=3&u=5",
  hash: "#abc"
};

const url = URL.format(obj);
console.log(url);
```

## 四、util 模块

提供实用工具函数。

### 常用 API

- `util.promisify(original)` — 将回调风格的函数转换为 Promise
- `util.callbackify(original)` — 将 Promise 风格的函数转换为回调
- `util.isDeepStrictEqual(val1, val2)` — 深度严格相等比较

```js
// promisify: 回调风格转 Promise
const util = require("util");

function delayCallBack(duration, callback) {
  setTimeout(() => {
    callback(null, duration);
  }, duration);
}

const delay = util.promisify(delayCallBack);
(async () => {
  const r = await delay(500);
  console.log(r); // 500
})();
```

```js
// callbackify: Promise 转回调
const util = require("util");

async function delay(duration = 1000) {
  return new Promise(resolve => {
    setTimeout(() => {
      resolve(duration);
    }, duration);
  });
}

const delayCallback = util.callbackify(delay);
delayCallback(500, (err, d) => {
  console.log(d); // 500
});
```

```js
// isDeepStrictEqual: 深度比较
const util = require("util");
const obj1 = { a: 1, b: { c: 3, d: { e: 5 } } };
const obj2 = { a: 1, b: { c: 3, d: { e: 5, g: 6 } } };
console.log(util.isDeepStrictEqual(obj1, obj2)); // false
```

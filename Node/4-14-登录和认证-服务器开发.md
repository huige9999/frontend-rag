# 4-14 登录和认证-服务器开发

## 知识要点

- 使用 `jsonwebtoken` 库实现 JWT 签发和验证
- 封装 JWT 工具模块（`routes/jwt.js`）
- 登录时签发 JWT，通过 cookie 和 authorization header 传递
- 认证中间件通过 JWT 验证用户身份
- 新增 `/api/admin/whoami` 接口用于获取当前登录用户信息

## JWT 工具模块

封装 JWT 的签发和验证逻辑：

```js
// routes/jwt.js
const secrect = "yuanjin";
const cookieKey = "token";
const jwt = require("jsonwebtoken");

// 颁发 jwt
exports.publish = function (res, maxAge = 3600 * 24, info = {}) {
  const token = jwt.sign(info, secrect, {
    expiresIn: maxAge,
  });
  // 添加到 cookie
  res.cookie(cookieKey, token, {
    maxAge: maxAge * 1000,
    path: "/",
  });
  // 添加到 authorization header
  res.header("authorization", token);
};

// 验证 jwt
exports.verify = function (req) {
  let token;
  // 尝试从 cookie 中获取
  token = req.cookies[cookieKey];
  if (!token) {
    // 尝试从 header 中获取
    token = req.headers.authorization;
    if (!token) {
      return null;
    }
    // authorization: bearer token
    token = token.split(" ");
    token = token.length === 1 ? token[0] : token[1];
  }
  try {
    const result = jwt.verify(token, secrect);
    return result;
  } catch (err) {
    return null;
  }
};
```

## 登录接口 - 签发 JWT

```js
// routes/api/admin.js
const jwt = require("../jwt");

router.post("/login", asyncHandler(async (req, res) => {
  const result = await adminServ.login(req.body.loginId, req.body.loginPwd);
  if (result) {
    let value = result.id;
    // 登录成功，签发 JWT
    jwt.publish(res, undefined, { id: value });
  }
  return result;
}));

// 新增 whoami 接口
router.get("/whoami", asyncHandler(async (req, res) => {
  return await adminServ.getAdminById(req.userId);
}));
```

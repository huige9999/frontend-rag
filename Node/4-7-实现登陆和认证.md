# 4-7 实现登录和认证

## 知识要点

- 使用 `cookie-parser` 中间件解析 Cookie
- 使用 `res.cookie()` 方法设置 Cookie
- 使用 token 中间件进行身份认证
- 对 token 进行加密（使用 `crypt` 模块）
- 使用 `path-to-regexp` 匹配需要认证的 API

## cookie-parser 中间件

加入后会在 `req` 对象中注入 `cookies` 属性获取所有 cookie，在 `res` 对象中注入 `cookie` 方法设置 cookie：

```js
const cookieParser = require("cookie-parser");
app.use(cookieParser());
```

## 登录接口 - 设置加密 Cookie

```js
// routes/api/admin.js
const cryptor = require("../../util/crypt");

router.post("/login", asyncHandler(async (req, res) => {
  const result = await adminServ.login(req.body.loginId, req.body.loginPwd);
  if (result) {
    let value = result.id;
    value = cryptor.encrypt(value.toString()); // 加密用户ID
    // 登录成功，通过 res.cookie 设置
    res.cookie("token", value, {
      path: "/",
      domain: "localhost",
      maxAge: 7 * 24 * 3600 * 1000, // 毫秒数
    });
    res.header("authorization", value); // 同时通过 header 传递
  }
  return result;
}));
```

## Token 认证中间件

```js
// routes/tokenMiddleware.js
const { pathToRegexp } = require("path-to-regexp");
const cryptor = require("../util/crypt");

// 需要认证的 API 列表
const needTokenApi = [
  { method: "POST", path: "/api/student" },
  { method: "PUT", path: "/api/student/:id" },
  { method: "GET", path: "/api/student" },
];

module.exports = (req, res, next) => {
  const apis = needTokenApi.filter((api) => {
    const reg = pathToRegexp(api.path);
    return api.method === req.method && reg.test(req.path);
  });
  if (apis.length === 0) {
    next();
    return;
  }

  // 从 cookie 或 header 中获取 token
  let token = req.cookies.token;
  if (!token) {
    token = req.headers.authorization;
  }
  if (!token) {
    handleNonToken(req, res, next);
    return;
  }
  const userId = cryptor.decrypt(token); // 解密获取用户ID
  req.userId = userId;
  next();
};

function handleNonToken(req, res, next) {
  res.status(403).send(getErr("you dont have any token to access the api", 403));
}
```

## 认证流程

1. 客户端 POST `/api/admin/login` 登录
2. 服务器验证成功后设置加密 cookie + authorization header
3. 后续请求自动附带 cookie，或手动附带 authorization header
4. tokenMiddleware 拦截需要认证的 API，验证 token
5. 验证通过后 `req.userId` 可用于后续业务逻辑

# 微信API—录音功能

本节课程使用微信小程序的录音管理器（`wx.getRecorderManager`）实现录音功能，配合文件系统（`wx.getFileSystemManager`）保存录音文件，以及音频上下文（`wx.createInnerAudioContext`）播放录音。

## 核心知识点

- `wx.getRecorderManager`：获取全局唯一的录音管理器
- `recorderManager.start()` / `recorderManager.stop()`：开始/停止录音
- `recorderManager.onStop`：监听录音停止事件，获取临时文件路径
- `wx.getFileSystemManager`：文件系统管理，用于保存和获取录音文件
- `wx.createInnerAudioContext`：创建音频上下文用于播放录音
- `bindtouchstart` / `bindtouchend`：实现按住录音、松开停止的交互

## 页面逻辑（index.js）

```javascript
const { formateDate } = require("../../utils/util");

let _this = null;    // 存储 this 指向
let timer = null;    // 用来停止计时器函数

// 播放音频的全局上下文对象
const audioContext = wx.createInnerAudioContext();
audioContext.autoplay = true;
audioContext.onEnded(()=>{
  wx.showToast({
    title: '播放结束',
    icon : "success",
    duration: 1000
  })
})

// 全局文件管理
const fileSystem = wx.getFileSystemManager();

// 全局的录音管理对象
const recoderManager = wx.getRecorderManager();
// 录音停止时触发，将临时文件保存到本地
recoderManager.onStop(res=>{
  const { tempFilePath } = res;
  fileSystem.saveFile({
    tempFilePath,
    success(res){
      console.log("savedFilePath:", res.savedFilePath);
      wx.showToast({
        title: '恭喜！录音成功',
        icon : "success",
        duration: 1000
      });
      getList();
    }
  })
});

/**
 * 从文件系统里面获取所有的录音记录
 */
function getList(){
  fileSystem.getSavedFileList({
    success(res){
      const voicesArr = [];
      res.fileList.sort((a,b) => a.createTime - b.createTime);
      for(let i=0;i<res.fileList.length;i++){
        const createTime = formateDate(res.fileList[i].createTime * 1000);
        const size = (res.fileList[i].size / 1024).toFixed(2);
        voicesArr.push({
          filePath : res.fileList[i].filePath,
          createTime,
          size
        })
      }
      _this.setData({
        voices : voicesArr
      })
    }
  })
}

/**
 * 麦克风动画（不停切换图片）
 */
function speaking(){
  let i = 1;
  timer = setInterval(()=>{
    i++;
    i %= 5;
    _this.setData({
      j : i
    })
  },200)
}

Page({
  data: {
    isSpeaking : false, // 是否显示录音图片
    j : 1,             // 录音图片的下标
    voices : [],       // 存储所有的录音
  },
  onLoad(){
    _this = this;
    getList();
  },
  // 按下的时候开始录音
  touchdown(){
    this.setData({
      isSpeaking : true
    });
    speaking(); // 播放麦克风动画
    recoderManager.start();
  },
  // 抬起来的时候停止录音
  touchup(){
    this.setData({
      isSpeaking : false
    });
    clearInterval(timer);
    timer = null;
    recoderManager.stop();
  },
  // 播放录音
  gotoPlay(e){
    audioContext.src = e.currentTarget.dataset.key;
    audioContext.play();
    wx.showToast({
      title: '开始播放',
      icon : "success",
      duration: 1000
    })
  },
  // 清除所有录音
  clearList(){
    fileSystem.getSavedFileList({
      success({fileList}){
        for(let i=0;i<fileList.length;i++){
          fileSystem.removeSavedFile({
            filePath : fileList[i].filePath,
            success(){
              if(i === fileList.length - 1){
                wx.showToast({
                  title: '已清除',
                  icon: "success",
                  duration: 1000
                });
                _this.setData({
                  voices : []
                })
              }
            }
          })
        }
      }
    })
  }
})
```

## 小结

本节实现了一个完整的录音应用，涉及三个核心模块：
1. **录音**：通过 `wx.getRecorderManager()` 获取录音管理器，`start()` 开始录音，`stop()` 停止录音。使用 `bindtouchstart` / `bindtouchend` 实现"按住录音、松开停止"的交互
2. **文件管理**：录音产生的是临时文件，需要通过 `wx.getFileSystemManager().saveFile()` 保存为永久文件。使用 `getSavedFileList` 获取所有录音列表，`removeSavedFile` 删除录音
3. **播放**：使用 `wx.createInnerAudioContext()` 创建音频上下文播放录音文件
4. **动画**：通过定时器不断切换图片下标，实现录音中的麦克风动画效果
